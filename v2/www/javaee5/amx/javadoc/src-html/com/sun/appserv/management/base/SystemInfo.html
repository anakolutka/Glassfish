030    package com.sun.appserv.management.base;
031    
032    
033    /**
034            Provides information about the capabilities of the running server.
035            Callers should check only for specific capabilities, never whether
036            the server is PE/SE/EE, since the feature assortment could vary with
037            release.
038     */
039    public interface SystemInfo extends AMX
040    {
041    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
042            public static final String      J2EE_TYPE                       = XTypes.SYSTEM_INFO;
043            
044            /**
045                    Call supportsFeature() with this value to determine if the server
046                    supports clusters.
047             */
048            public final String     CLUSTERS_FEATURE        = "SupportsClusters";
049            
050            /**
051                    Call supportsFeature() with this value to determine if the server
052                    supports more than one server.
053             */
054            public final String     MULTIPLE_SERVERS_FEATURE        = "SupportsMultipleServers";
055            
056            
057            /**
058                    Call supportsFeature() with this value to determine if this MBean
059                    is running in the Domain Admin Server.
060             */
061            public final String     RUNNING_IN_DAS_FEATURE          = "RunningInDomainAdminServer";
062            
063            
064            
065            /**
066                    Query whether a feature is supported.  Features require the use
067                    of a key, which may be any of:
068                    <ul>
069                    <li>CLUSTERS_FEATURE</li>
070                    <li>MULTIPLE_SERVERS_FEATURE</li>
071                    <li>RUNNING_IN_DAS_FEATURE</li>
072                    </ul>
073                    
074                    @param key      the feature to query
075             */
076            public boolean  supportsFeature( String key );
077            
078            /**
079                    Return all features names.
080                    
081                    @return Set
082             */
083            public String[]         getFeatureNames();
084            
085    
086    
087    
088    }
