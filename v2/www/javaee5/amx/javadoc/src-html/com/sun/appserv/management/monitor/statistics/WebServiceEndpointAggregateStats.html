024    package com.sun.appserv.management.monitor.statistics;
025    
026    import javax.management.j2ee.statistics.Stats;
027    import javax.management.j2ee.statistics.CountStatistic;
028    import com.sun.appserv.management.j2ee.statistics.StringStatistic;
029    import com.sun.appserv.management.j2ee.statistics.NumberStatistic;
030    
031    /**
032     * Web Service Endpoint's stats interface. It provides faults, response time,
033     * throughput and authentication failure/success information.
034     *
035     * @since AppServer 9.0
036     */
037    public interface WebServiceEndpointAggregateStats extends Stats
038    {
039        /**
040         * Returns the total number of fault as a CountStatistic.
041         * Generally if an Endpoint results in a fault, this count will
042         * increment by one.
043         *
044         * @return an instance of {@link CountStatistic}
045         */
046        public CountStatistic getTotalFaults();
047    
048        /**
049         * Returns the total number of successful runs, as a CountStatistic.
050         * Generally if an operation returns with out a fault it is consider a
051         * success. normally, this count will increment by one.
052         *
053         * @return an instance of {@link CountStatistic}
054         */
055        public CountStatistic getTotalNumSuccess();
056    
057        /**
058         * Returns the average time in milli seconds spent during the last 
059         * successful/unsuccessful attempt to execute the operation, as a
060         * CountStatistic. The time spent is generally an indication of 
061         * the system load/processing time.
062         *
063         * @return an instance of {@link CountStatistic}
064         */
065        public CountStatistic getAverageResponseTime();
066    
067        /**
068         * Returns the time in milli seconds spent during the last 
069         * successful/unsuccessful attempt to execute the operation, as a
070         * CountStatistic.  
071         *
072         * @return an instance of {@link CountStatistic}
073         */
074        public CountStatistic getResponseTime();
075    
076        /**
077         * Returns the minimum time spent in milli seconds for any successful/
078         * unsuccessful attempt to execute the operation, as a CountStatistic.
079         *
080         * @return an instance of {@link CountStatistic}
081         */
082        public CountStatistic getMinResponseTime();
083    
084        /**
085         * Returns the maximum time spent in milli seconds for any successful/
086         * unsuccessful attempt to execute the operation, as a CountStatistic.
087         *
088         * @return an instance of {@link CountStatistic}
089         */
090        public CountStatistic getMaxResponseTime();
091    
092        /**
093         * Returns the number successful messages/minute since the server is
094         * started as a NumberStatistic.
095         *
096         * @return an instance of {@link NumberStatistic}
097         */
098        public NumberStatistic getThroughput();
099    
100        /**
101         * Returns the total number of authentication failures as a
102         * CountStatistic.
103         *
104         * @return an instance of {@link CountStatistic}
105         */
106        public CountStatistic getTotalAuthFailures();
107    
108        /**
109         * Returns the total number of authentication successes
110         * CountStatistic.
111         *
112         * @return an instance of {@link CountStatistic}
113         */
114        public CountStatistic getTotalAuthSuccesses();
115    }
