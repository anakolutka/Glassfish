005    package amxsamples;
006    
007    import javax.net.ssl.HandshakeCompletedEvent;
008    
009    import com.sun.appserv.management.client.HandshakeCompletedListenerImpl;
010    
011    
012    /**
013            Demonstrates how to write a HandshakeCompletedListener. Note that usually
014            it is not necessary to write your own HandshakeCompletedListener since the
015            event is available from {@link HandshakeCompletedListenerImpl#getLastEvent}.
016            <p>
017            You may wish to write a HandshakeCompletedListener if the data contained
018            in the HandshakeCompletedEvent is of interest or you wish to exert more
019            control over the TLS connection.
020     */
021    public final class SampleHandshakeCompletedListener
022            extends HandshakeCompletedListenerImpl
023    {
024                    public
025            SampleHandshakeCompletedListener()
026            {
027            }
028            
029                    public synchronized void
030            handshakeCompleted( final HandshakeCompletedEvent event) 
031            {
032                    super.handshakeCompleted( event );
033                    
034                    System.out.println( "HandshakeCompleted:\n" + event + "\n" );
035            }
036    }
