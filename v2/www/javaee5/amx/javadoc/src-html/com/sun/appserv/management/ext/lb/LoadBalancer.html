030    package com.sun.appserv.management.ext.lb;
031    
032    import java.util.Date;
033    
034    import static com.sun.appserv.management.base.XTypes.*;
035    import com.sun.appserv.management.base.AMX;
036    
037    /**
038       Runtime counterpart for the config MBean {@link com.sun.appserv.management.config.LoadBalancerConfig} 
039       representing the load-balancer element.
040     */
041    public interface LoadBalancer extends AMX {
042    
043        /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
044        public static final String  J2EE_TYPE = LOAD_BALANCER;
045    
046        /**
047          Applies changes in the corresponding configuration to this LB 
048         */
049        public void applyLBChanges();        
050    
051        /**
052          @return true if there are pending changes for this LB
053         */
054        public boolean isApplyChangeRequired();       
055    
056        /**
057          Exports the corresponding LBConfig information and returns the contents as a string.
058          @see com.sun.appserv.management.config.LBConfig
059         */
060        public String getLoadBalancerXML();    
061          
062        /**
063          Returns the timestamp of the most recent export of corresponding LBConfig
064         */
065        public Date getLastExported();        
066        
067        /**
068          Returns the timestamp of the most recent application of corresponding LBConfig
069         */
070        public Date getLastApplied();            
071    
072        /**
073          Returns the timestamp of the last time the stats on this loadbalancer were reset
074         */
075        public Date getLastResetTime();            
076    
077        /**
078          Reset the monitoring stats on this loadbalancer.
079         */
080        public void resetStats();
081        
082        /**
083          Reset the monitoring stats on this loadbalancer.
084         */
085        public boolean testConnection();
086                
087        /**
088         * Returns the uhealthy/healthy/quiesced status for an insatnce load balanced
089         * by this load balancer.
090         */
091        public String getStatus(String instanceName);     
092    }
