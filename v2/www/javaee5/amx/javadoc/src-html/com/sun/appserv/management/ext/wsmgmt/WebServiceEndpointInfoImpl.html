032    /**
033     * This is the mechanism to provide web service endpoint's information in a
034     * given module. 
035     *
036     * @since AppServer 9.0
037     */
038    public final class WebServiceEndpointInfoImpl
039        extends MapCapableBase
040        implements WebServiceEndpointInfo {
041    
042        /**
043         * Public constructor
044         * 
045         * @param m         Map containing keys as in {@link WebServiceEndpointInfo}
046         *                  and their values
047         * @param className Class name of the interface , must be
048         *                  WebServiceEndpointInfo.CLASS_NAME
049         */
050        public WebServiceEndpointInfoImpl(
051            final Map<String,Serializable> m,
052            final String className) {
053            super(m, className);
054        }
055    
056        /**
057         * Public constructor
058         * 
059         * @param name            Name of the web service end point
060         * @param uri             URI of the web service end point
061         * @param appid           Application Id of this web service end point
062         * @param bundleName      Bundle name (module name) of this web service end
063         *                        point
064         * @param isStandalone    true, if the module (ejb or web) of this web 
065         *                        service end point is standalone
066         * @param wsdl            WSDL file location for this web service end point
067         * @param mapfile         Mapping file location for this web service end 
068         *                        point
069         * @param webservicesFile webservices.xml file location for this web 
070         *                        service end point
071         * @param implType        Implementation type of this web service end point
072         * @param implName        Implemenation Ejb or Servlet name
073         * @param implClass       Implementation Ejb or Servlet class name
074         */
075        public WebServiceEndpointInfoImpl( String name, String uri, String appid, 
076            String bundleName, boolean isStandalone, String wsdl, String mapfile, 
077            String webservicesFile, String implType, String implName, 
078            String implClass) {
079    
080            super (null, WebServiceEndpointInfo.CLASS_NAME);
081            setName(name);
082            setEndpointURI(uri);
083            setAppID(appid);
084            setIsAppStandaloneModule(isStandalone);
085            setBundleName(bundleName);
086            setWSDLFile(wsdl);
087            setWebservicesFile(webservicesFile);
088            setMappingFile(mapfile);
089            setServiceImplType(implType);
090            setServiceImplName(implName);
091            setServiceImplClass(implClass);
092        }
093    
094        protected boolean validate() {
095            return (true);
096        }
097    
098        /**
099         * Returns the name of the WebService.
100         *
101         * @return fully qualified name of this WebService

103        public String getName() {

105        }
106    
107        /**
108         * Sets the name of the WebService.
109         */

111            putField(WebServiceEndpointInfo.NAME_KEY, name);
112        }

121        }

126        public void setEndpointURI(String uri) {
127            putField(WebServiceEndpointInfo.END_POINT_URI_KEY, uri);
128        }

138        }

149         * Returns true if this web service belongs to a stand alone module.
150         *

154            return getBoolean(WebServiceEndpointInfo.IS_STAND_ALONE_MODULE_KEY).booleanValue();

163        public void setIsAppStandaloneModule(boolean isAppStandAloneModule) {

166        }

391        /**
392         * Returns the descriptor . The descriptors include WSDL, 
393         * webservices.xml and mapping file. If the alt WSDL is specified
394         * in the archive, the overridden (correct) WSDL file is returned.
395         *
396         * @return the descriptors 
397         */
398        public String[] getDescriptors() {
399            String [] list = new String[3];
400            list[0] = getString(WebServiceEndpointInfo.WSDL_FILE_KEY);
401            list[1] = getString(WebServiceEndpointInfo.WEB_SERVICES_FILE_KEY);
402            list[2] = getString(WebServiceEndpointInfo.MAPPING_FILE_KEY);
403            return list;
404        }
