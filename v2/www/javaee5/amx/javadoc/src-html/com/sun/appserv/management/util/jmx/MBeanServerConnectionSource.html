030    package com.sun.appserv.management.util.jmx;
031    
032    import javax.management.MBeanServer;
033    import javax.management.MBeanServerConnection;
034    import javax.management.remote.JMXConnector;
035    
036    import com.sun.appserv.management.client.ConnectionSource;
037    
038    
039    /**
040            A ConnectionSource for in-process access where the MBeanServer is actually
041            known and where later access to it may be desired.
042     */
043    public final class MBeanServerConnectionSource implements ConnectionSource
044    {
045            protected MBeanServer           mServer;
046            
047            /**
048                    Create a new instance for the specified MBeanServer
049             */
050                    public
051            MBeanServerConnectionSource( final MBeanServer server )
052            {
053                    mServer = server;
054            }
055            
056            /**
057                    Return the internal MBeanServer 
058             */
059                    public MBeanServer
060            getMBeanServer( )
061            {
062                    return( mServer );
063            }
064            
065                    public MBeanServerConnection
066            getExistingMBeanServerConnection( )
067            {
068                    return( getMBeanServer() );
069            }
070            
071                    public MBeanServerConnection
072            getMBeanServerConnection( boolean forceNew )
073            {
074                    return( getExistingMBeanServerConnection() );
075            }
076            
077            
078                    public JMXConnector
079            getJMXConnector( boolean forceNew )
080            {
081                    // we can't supply one...
082                    return( null );
083            }
084    }
