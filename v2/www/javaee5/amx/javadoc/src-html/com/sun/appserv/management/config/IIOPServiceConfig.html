031    package com.sun.appserv.management.config;
032    
033    import java.util.Map;
034    
035    import com.sun.appserv.management.base.XTypes;
036    import com.sun.appserv.management.base.Container;
037    
038    
039    
040    
041    
042    
043    /**
044             Configuration for the &lt;iiop-service&gt; element.
045     */
046    public interface IIOPServiceConfig
047            extends ConfigElement, Container
048    {
049    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
050            public static final String      J2EE_TYPE       = XTypes.IIOP_SERVICE_CONFIG;
051    
052            /**
053                    Creates ssl-client-config element.
054    
055                    @param params
056                    @return Returns a proxy to the SSLConfig MBean.
057             */
058            public SSLConfig        createIIOPSSLClientConfig( Map<String,String> params );
059    
060            /**
061                    Removes ssl-client-config element.
062             */
063            public void                     removeIIOPSSLClientConfig();
064    
065            /**
066                    Get the SSLConfig MBean that corresponds to ssl-client-config
067                    element of this iiop service.
068             */
069            public SSLConfig        getIIOPSSLClientConfig();
070    
071            public boolean          getClientAuthenticationRequired();
072    
073            public void             setClientAuthenticationRequired( final boolean value );
074    
075    
076    
077            /**
078                    Calls Container.getContaineeMap( XTypes.IIOP_LISTENER_CONFIG ).
079                    @return Map of IIOPListenerConfig MBean proxies, keyed by name.
080                    @see com.sun.appserv.management.base.Container#getContaineeMap
081             */
082            public Map<String,IIOPListenerConfig>             getIIOPListenerConfigMap();
083            
084            
085            /**
086                    Create an <iiop-listener>. Most fields will default to reasonable values.
087                    Caller should modify them after creation, if desired.
088                    
089                    @param name     the name (id) of the newly created listener
090                    @param address  IP address of the listener
091                    @param optional optional parameters
092                    @see IIOPListenerConfigKeys
093             */
094            public IIOPListenerConfig       createIIOPListenerConfig( String name, String address, Map<String,String> optional );
095    
096            /**
097                    Removes an iiop listener.
098    
099                    @param name the name (id) of the iiop listener to be removed.
100             */
101            public void                     removeIIOPListenerConfig( String name );
102    
103    
104            /**
105                    Get a proxy to an ORBConfig
106             */
107            ORBConfig                                       getORBConfig();
108    }
