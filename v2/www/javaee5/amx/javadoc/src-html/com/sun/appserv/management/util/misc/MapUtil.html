077            /**
078                    Create a new Map consisting of a single key/value pair.
079             */
080                    public static <K,V> Map<K,V>
081            newMap( 
082                    final Map<K,V>    m1,
083                    final Map<K,V>    m2 )
084            {
085                    final Map<K,V>    m       = new HashMap<K,V>();
086                    
087                    if ( m1 != null )
088                    {
089                            m.putAll( m1 );
090                    }
091                    if ( m2 != null )
092                    {
093                            m.putAll( m2 );
094                    }
095                    
096                    return( m );
097            }

099            /**
100                    Create a new Map and insert the specified mappings as found in 'mappings'.
101                    The even-numbered entries are the keys, and the odd-numbered entries are
102                    the values.
103             */
104                    public static <T> Map<T,T>
105            newMap( final T[] mappings )
106            {
107                    if ( (mappings.length % 2) != 0 )
108                    {
109                            throw new IllegalArgumentException( "mappings must have even length" );
110                    }
111                    
112                    final Map<T,T> m  = new HashMap<T,T>();
113                    
114                    for( int i = 0; i < mappings.length; i += 2 )
115                    {
116                            m.put( mappings[ i ], mappings[ i + 1 ] );
117                    }
118                    
119                    return( m );
120            }

160                    public static boolean
161            mapsEqual(
162                    final Map<?,?>    m1,
163                    final Map<?,?>    m2 )
164            {
165                    if ( m1 == m2 )
166                    {
167                            return( true );
168                    }
169                    
170                    boolean equal   = false;
171                    
172                    if ( m1.size() == m2.size() &&
173                            m1.keySet().equals( m2.keySet() ) )
174                    {
175                            equal   = true;
176                            
177                            for( final Object key : m1.keySet() )
178                            {
179                                    final Object    value1  = m1.get( key );
180                                    final Object    value2  = m2.get( key );
181                                    
182                                    if ( ! CompareUtil.objectsEqual( value1, value2 ) )
183                                    {
184                                            equal   = false;
185                                            break;
186                                    }
187                            }
188                    }
189                    
190                    return( equal );
191            }

194                    public static <K,V> Map<K,V>
195            newMapNoNullValues( final Map<K,V> m )
196            {
197                    final Map<K,V>    result  = new HashMap<K,V>();
198                    
199                    for ( final K key : m.keySet() )
200                    {
201                            final V value   = m.get( key );
202                            
203                            if ( value != null )
204                            {
205                                    result.put( key, value );
206                            }
207                    }
208                    
209                    return( result );
210            }

218                    public static String
219            toString( final Map<?,?> m, final String separator )
220            {
221                    if ( m == null )
222                    {
223                            return( "null" );
224                    }
225                    
226                    final StringBuffer      buf     = new StringBuffer();
227                    
228                    final String[]  keyStrings  = getKeyStrings( m );
229                    for( final String key : keyStrings )
230                    {
231                            final Object    value   = m.get( key );
232                            
233                            buf.append( key );
234                            buf.append( "=" );
235                            buf.append( SmartStringifier.toString( value ) );
236                            buf.append( separator );
237                    }
238                    if ( buf.length() != 0 )
239                    {
240                            // strip trailing separator
241                            buf.setLength( buf.length() - separator.length() );
242                    }
243                    
244                    return( buf.toString() );
245            }

247                    public static <K> Set<K>
248            getNullValueKeys( final Map<K,?> m)
249            {
250                    final Set<K>      s       = new HashSet<K>();
251                    
252                    for( final K key : m.keySet() )
253                    {
254                            if ( m.get( key ) == null )
255                            {
256                                    s.add( key );
257                            }
258                    }
259                    return( s );
260            }

278            /**
279                @return true if non-null Map and all keys and values are of type java.lang.String
280             */
281                public static boolean
282            isAllStrings( final Map<?,?> m )
283            {
284                return m != null && CollectionUtil.isAllStrings( m.keySet() ) &&
285                        CollectionUtil.isAllStrings( m.values() );
286            }

288            /**
289                Convert an arbitrary Map to one whose keys and values
290                are both of type String.
291             */
292                public static Map<String,String>
293            toStringStringMap( final Map<?,?> m )
294            {
295                if ( m == null )
296                {
297                    return null;
298                }
299                
300                Map<String,String>  result  = null;
301                
302                if ( isAllStrings( m ) )
303                {
304                    result  = TypeCast.asMap( m );
305                }
306                else
307                {
308                    result = new HashMap<String,String>();
309                    
310                for( final Object key : m.keySet() )
311                {
312                    final Object value = m.get( key );
313                    
314                    if ( (key instanceof String) && (value instanceof String) )
315                    {
316                        result.put( (String)key, (String)value );
317                    }
318                    else
319                    {
320                        result.put( "" + key, "" + value );
321                    }
322                }
323                }
324                
325                return result;
326            }
