154            /**
155                    Build an ObjectName pattern.
156                    
157                    @param domain   the JMX domain
158                    @param props    properties of the ObjectName
159             */
160                    public static ObjectName
161            newObjectNamePattern(
162                    final String domain,
163                    final String props )
164            {
165                    String  actualProps     = null;
166                    
167                    if ( props.endsWith( JMXUtil.WILD_PROP ) ||
168                            props.equals( JMXUtil.WILD_ALL ) )
169                    {
170                            actualProps     = props;
171                    }
172                    else if ( props.length() == 0 )
173                    {
174                            actualProps     = "*";
175                    }
176                    else
177                    {
178                            actualProps     = props + WILD_PROP;
179                    }
180                    
181                    return( newObjectName( domain + ":" + actualProps ) );
182            }

207                    public static ObjectName
208            removeProperty(
209                    final ObjectName        objectName,
210                    final String            key )
211            {
212                    ObjectName      nameWithoutKey  = objectName;
213                    
214                    if ( objectName.getKeyProperty( key ) != null )
215                    {
216                            final String                            domain  = objectName.getDomain();
217                            final java.util.Hashtable       props   = objectName.getKeyPropertyList();
218                            
219                            props.remove( key );
220                            
221                            if ( objectName.isPropertyPattern() )
222                            {
223                                    final String    propsString     = mapToProps( props );
224                            
225                                    nameWithoutKey  = newObjectNamePattern( domain,
226                                                    nameWithoutKey.getKeyPropertyListString() );
227                            }
228                            else
229                            {
230                                    try
231                                    {
232                                        nameWithoutKey      = new ObjectName( domain, props );
233                                    }
234                                    catch( Exception e )
235                                    {
236                                        throw new RuntimeException( e );
237                                    }
238                            }
239                    }
240                    
241                    return( nameWithoutKey );
242            }

261                    public static String
262            concatenateProps( String props1, String props2 )
263            {
264                    String  result  = null;
265                    
266                    if ( props1.length() == 0 )
267                    {
268                            result  = props2;
269                    }
270                    else if ( props2.length() == 0 )
271                    {
272                            result  = props1;
273                    }
274                    else
275                    {
276                            result  = props1 + "," + props2;
277                    }
278                    
279                    return( result );
280            }

351                    public static String
352            getProps(
353                    final ObjectName        objectName,
354                    final Set<String> propKeys,
355                    final boolean           ignoreMissing )
356            {
357                    String  props   = "";
358                    
359                    final Iterator  iter    = propKeys.iterator();
360                    while( iter.hasNext() )
361                    {
362                            final String    key     = (String)iter.next();
363                            
364                            final String    pair    = getProp( objectName, key );
365                            if ( pair != null )
366                            {
367                                    props   = concatenateProps( props, pair );
368                            }
369                            else if ( ! ignoreMissing )
370                            {
371                                    throw new IllegalArgumentException(
372                                            "key not found: " + key + " in " + objectName );
373                            }
374                    }
375                    return( props );
376            }

412                    @param candidateKeys

420            {

422                    
423                    String  match   = null;
424                    
425                    while ( iter.hasNext() )
426                    {
427                            final String    key     = (String)iter.next();
428                            
429                            if ( objectName.getKeyProperty( key ) != null )

431                                    match   = key;

433                            }
434                    }
435                    
436                    return( match );

453            {
454                    final Set<ObjectName>     result  = new HashSet<ObjectName>();

719                    public static MBeanInfo
720            removeAttributes(
721                    final MBeanInfo origInfo,
722                    final String[]  attributeNames )
723            {
724                    MBeanInfo       result  = origInfo;
725                    
726                    if ( attributeNames.length != 0 )
727                    {
728                            final Map       infos   = JMXUtil.attributeInfosToMap( origInfo.getAttributes() );
729                            
730                            for( int i = 0; i < attributeNames.length; ++i )
731                            {
732                                    infos.remove( attributeNames[ i ] );
733                            }
734                            
735                            final MBeanAttributeInfo[]      newInfos = new MBeanAttributeInfo[ infos.keySet().size() ];
736                            infos.values().toArray( newInfos );
737                            
738                            result  = new MBeanInfo(
739                                            origInfo.getClassName(),
740                                            origInfo.getDescription(),
741                                            newInfos,
742                                            origInfo.getConstructors(),
743                                            origInfo.getOperations(),
744                                            origInfo.getNotifications() );
745                    }
746            
747                    return( result );
748            }

882                    private static AttributeList
883            getAttributesSingly(
884                    MBeanServerConnection   conn,
885                    ObjectName                              objectName,
886                    String[]                                attrNames,
887                    Set<String>                               problemNames )
888                    throws InstanceNotFoundException
889            {
890                    AttributeList   attrs   = new AttributeList();
891                    
892                    for( int i = 0; i < attrNames.length; ++i )
893                    {
894                            final String    name    = attrNames[ i ];
895                            
896                            try
897                            {
898                                    final Object    value   = conn.getAttribute( objectName, name );
899                                    
900                                    attrs.add( new Attribute( name, value ) );
901                            }
902                            catch( Exception e )
903                            {
904                                    // if the MBean disappeared while processing, just consider it gone
905                                    // from the start, even if we got some Attributes
906                                    if ( e instanceof InstanceNotFoundException )
907                                    {
908                                            throw (InstanceNotFoundException)e;
909                                    }
910                                    
911                                    if ( problemNames != null )
912                                    {
913                                            problemNames.add( name );
914                                    }
915                            }
916                    }
917                    
918                    return( attrs );
919            }

922            /**
923                    Get the Attributes using getAttributes() if possible, but if exceptions
924                    are encountered, attempt to get them one-by-one.
925                    
926                    @param conn                     the conneciton
927                    @param objectName       name of the object to access
928                    @param attrNames        attribute names
929                    @param problemNames     optional Set to which problem names will be added.
930                    @return AttributeList
931             */
932                    public static AttributeList
933            getAttributesRobust(
934                    MBeanServerConnection   conn,
935                    ObjectName                              objectName,
936                    String[]                                attrNames,
937                    Set<String>                               problemNames )
938                    throws InstanceNotFoundException, IOException
939            {
940                    AttributeList   attrs   = null;
941                    
942                    if ( problemNames != null )
943                    {
944                            problemNames.clear();
945                    }
946                    
947                    try
948                    {
949                            attrs   = conn.getAttributes( objectName, attrNames );
950                            if ( attrs == null )
951                            {
952                                    attrs   = new AttributeList();
953                            }
954                    }
955                    catch( InstanceNotFoundException e )
956                    {
957                            // if it's not found, we can't do anything about it.
958                            throw e;
959                    }
960                    catch( IOException e )
961                    {
962                            if ( connectionIsDead( conn ) )
963                            {
964                                    throw e;
965                            }
966                            
967                            // connection is still good
968                            
969                            attrs   = getAttributesSingly( conn, objectName, attrNames, problemNames );
970                    }
971                    catch( Exception e )
972                    {
973                            attrs   = getAttributesSingly( conn, objectName, attrNames, problemNames );
974                    }
975                    
976                    return( attrs );
977            }

981             */
982            public static final class MBeanOperationInfoComparator implements java.util.Comparator
983            {
984                    private static final MBeanOperationInfoStringifier              OPERATION_INFO_STRINGIFIER      =
985                            new MBeanOperationInfoStringifier( new MBeanFeatureInfoStringifierOptions( false, ",") );
986                            
987                            
988                    public static final MBeanOperationInfoComparator        INSTANCE        = new MBeanOperationInfoComparator();
989                    
990                    private MBeanOperationInfoComparator()  {}
991                    
992                            public int
993                    compare( Object o1, Object o2 )
994                    {
995                            final MBeanOperationInfoStringifier     sf      = OPERATION_INFO_STRINGIFIER;
996                            
997                            final MBeanOperationInfo        info1   = (MBeanOperationInfo)o1;
998                            final MBeanOperationInfo        info2   = (MBeanOperationInfo)o2;
999                            
1000                            // we just want to sort based on name and signature; there can't be two operations with the
1001                            // same name and same signature, so as long as we include the name and signature the
1002                            // sorting will always be consistent.
1003                            int     c       = info1.getName().compareTo( info2.getName() );
1004                            if ( c == 0 )
1005                            {
1006                                    // names the same, subsort on signature, first by number of params
1007                                    c       = info1.getSignature().length - info2.getSignature().length;
1008                                    if ( c == 0 )
1009                                    {
1010                                            // names the same, subsort on signature, first by number of params
1011                                            c       = sf.getSignature( info1 ).compareTo( sf.getSignature( info2 ) );
1012                                    }
1013                                    
1014                            }
1015                            
1016                            return( c );
1017                    }
1018                    
1019                            public boolean
1020                    equals( Object other )
1021                    {
1022                            return( other instanceof MBeanOperationInfoComparator );
1023                    }

1056            /**
1057                    Return true if the two MBeanAttributeInfo[] contain the same attributes
1058                    WARNING: arrays will be sorted to perform the comparison if they are the same length.
1059             */
1060                    boolean
1061            sameAttributes( MBeanAttributeInfo[] infos1, MBeanAttributeInfo[] infos2 )
1062            {
1063                    boolean equal   = false;
1064                    
1065                    if( infos1.length == infos2.length )
1066                    {
1067                            equal   = ArrayUtil.arraysEqual( infos1, infos2 );
1068                            if ( ! equal )
1069                            {
1070                                    // could still be equal, just in different order
1071                                    Arrays.sort( infos1, MBeanAttributeInfoComparator.INSTANCE );
1072                                    Arrays.sort( infos2, MBeanAttributeInfoComparator.INSTANCE );
1073                                    
1074                                    equal   = true; // reset to false upon failure
1075                                    for( int i = 0; i < infos1.length; ++i )
1076                                    {
1077                                            if ( ! infos1[ i ].equals( infos2[ i ] ) )
1078                                            {
1079                                                    equal   = false;
1080                                                    break;
1081                                            }
1082                                    }
1083                            }
1084                            else
1085                            {
1086                                    equal   = true;
1087                            }
1088                    }
1089                    return( equal );
1090            }

1094                    Return true if the two MBeanAttributeInfo[] contain the same operations
1095                    WARNING: arrays will be sorted to perform the comparison if they are the same length.
1096             */
1097                    boolean
1098            sameOperations( MBeanOperationInfo[] infos1, MBeanOperationInfo[] infos2 )
1099            {
1100                    boolean equal   = false;
1101                    
1102                    if ( infos1.length == infos2.length )
1103                    {
1104                            // if they're in identical order, this is the quickest test if they ultimately succeed
1105                            equal   = ArrayUtil.arraysEqual( infos1, infos2 );
1106                            if ( ! equal )
1107                            {
1108                                    // could still be equal, just in different order
1109                                    Arrays.sort( infos1, MBeanOperationInfoComparator.INSTANCE );
1110                                    Arrays.sort( infos2, MBeanOperationInfoComparator.INSTANCE );
1111                                    
1112                                    equal   = true; // reset to false upon failure
1113                                    for( int i = 0; i < infos1.length; ++i )
1114                                    {
1115                                            if ( ! infos1[ i ].equals( infos2[ i ] ) )
1116                                            {
1117                                                    equal   = false;
1118                                                    break;
1119                                            }
1120                                    }
1121                            }
1122                    }
1123                    return( equal );
1124            }

1510                    
1511                    @param infos1
1512                    @param infos2

1515            mergeMBeanAttributeInfos(
1516                    final MBeanAttributeInfo[]      infos1,
1517                    final MBeanAttributeInfo[]      infos2 )
1518            {

1520                final Set<String>   names   = new HashSet<String>();
1521                for( final MBeanAttributeInfo info : infos1 )
1522                {

1525    
1526                    final Set<MBeanAttributeInfo>     merged  = SetUtil.newSet( infos1 );
1527                    
1528                    for( final MBeanAttributeInfo info2 : infos2 )
1529                    {
1530                            final String    info2Name       = info2.getName();
1531                            
1532                            if ( ! names.contains( info2Name ) )
1533                            {
1534                                    merged.add( info2 );
1535                            }
1536                    }
1537

1539                            new MBeanAttributeInfo[ merged.size() ];

1708                    public static boolean
1709            domainMatches(
1710                    final String            defaultDomain,
1711                    final ObjectName        pattern,
1712                    final ObjectName        candidate )
1713            {
1714                    boolean matches = false;
1715                    
1716                    final String    candidateDomain = candidate.getDomain();
1717                    if ( pattern.isDomainPattern() )
1718                    {
1719                            final String    regex   =
1720                                    RegexUtil.wildcardToJavaRegex( pattern.getDomain() );
1721                            
1722                            matches = Pattern.matches( regex, candidateDomain);
1723                    }
1724                    else
1725                    {       
1726                            // domain is not a pattern
1727                            
1728                            String  patternDomain   = pattern.getDomain();
1729                            if ( patternDomain.length() == 0 )
1730                            {
1731                                    patternDomain   = defaultDomain;
1732                            }
1733                            
1734                            matches = patternDomain.equals( candidateDomain );
1735                    }
1736                    
1737                    //dm( "MBeanProxyMgrImpl.domainMatches: " + matches + " " + pattern + " vs " + candidate );
1738                    
1739                    return( matches );
1740            }
1741            
1742                    public static boolean
1743            matchesPattern(
1744                    final String            defaultDomain,
1745                    final ObjectName        pattern,
1746                    final ObjectName        candidate )
1747            {
1748                    boolean matches = false;
1749                    
1750                    if ( domainMatches( defaultDomain, pattern, candidate ) )
1751                    {
1752                            final String    patternProps    = pattern.getCanonicalKeyPropertyListString();
1753                            final String    candidateProps  = candidate.getCanonicalKeyPropertyListString();
1754                            assert(  patternProps.indexOf( "*" ) < 0 );
1755                            assert(  candidateProps.indexOf( "*" ) < 0 );
1756                            
1757                            // Since we used canonical form any match means the pattern props String
1758                            // must be a substring of candidateProps
1759                            if ( candidateProps.indexOf( patternProps ) >= 0 )
1760                            {
1761                                    matches = true;
1762                            }
1763                    }
1764                    
1765                    return( matches );
1766            }
