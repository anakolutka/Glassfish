025    package com.sun.appserv.management.ext.logging;
026    
027    
028    import com.sun.appserv.management.base.AMX;
029    import com.sun.appserv.management.base.XTypes;
030    import com.sun.appserv.management.base.Utility;
031    import com.sun.appserv.management.monitor.Monitoring;
032    
033    /**
034            Supports accessing logging information in multiple ways.  The following are supported:
035            <ul>
036            <li>Emission of pseudo real-time JMX Notifications when a
037                    log record is created--see {@link LogRecordEmitter}</li>
038            <li>Access to existing log file contents--see {@link LogFileAccess}</li>
039            <li>Querying for log entries--see {@link LogQuery}</li>
040            </ul>
041            <p>
042            A Logging always has j2eeType={@link XTypes#LOGGING} and
043            the same name as the server it represents.
044            @since AS 9.0
045            @see com.sun.appserv.management.monitor.ServerRootMonitor#getLogging
046     */
047    public interface Logging
048            extends LogRecordEmitter, LogQuery, LogFileAccess, LogAnalyzer, AMX
049    {
050            public static final String      J2EE_TYPE                       = XTypes.LOGGING;
051            
052       /**
053                    Sets the log level of the Logger for the specified module.  This operation
054                    will not effect a change to the corresponding loggin configuration for that module.
055                    <b>This capability is subject to removal, due to the confusion
056                    caused by configuration settings not matching</b>.
057                    
058                    @param module   a module name as specified in {@link LogModuleNames}.
059                    @param level    a log level
060            @see com.sun.appserv.management.config.ModuleLogLevelsConfig
061         */
062        public void setModuleLogLevel( String module, String level );
063        
064       /**
065                    Gets the log level of the Logger for the specified module, which may or may not
066                    be the same as that found in the configuration.
067                    <b>This capability is subject to removal, due to the confusion
068                    caused by configuration settings not matching</b>.
069                    
070                    @param moduleName a module name as specified in {@link LogModuleNames}
071                    @see com.sun.appserv.management.config.ModuleLogLevelsConfig
072        */
073        public String getModuleLogLevel( String moduleName );
074        
075        /**
076            This method may be used to verify that your Logging listener is working
077            correctly.
078            @param level the log level of the log message.
079            @param message  the message to be placed in Notif.getMessage()
080         */
081        public void testEmitLogMessage( final String level, final String message );
082    }
