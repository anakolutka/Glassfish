023    package com.sun.appserv.management.client.prefs;
024    
025    /**
026     * An immutable class that represents an arbitrary LoginInfo for Appserver Administration Client. A LoginInfo
027     * is specific to an admin host and admin port. Thus, with this scheme, there can be
028     * at the most one LoginInfo for an operating system user of Appserver, for a given admin host
029     * and admin port.
030     * @since Appserver 9.0
031     */
032    public final class LoginInfo implements Comparable<LoginInfo> {
033        private String host;
034        private int    port;
035        private String user;
036        private String password;
037        
038        /**
039         * Creates an Immutable instance of a LoginInfo from given 4-tuple. 
040         * The host, user and password may not be null.
041         * The port may not be a negative integer.
042         * @param host String representing host
043         * @param port integer representing port
044         * @param user String representing user
045         * @param password String representing password
046         * @throws IllegalArgumentException if parameter contract is violated
047         */
048        public LoginInfo(final String host, final int port, final String user, final String password) {
049            if (host == null || port < 0 || user == null || password == null)
050                throw new IllegalArgumentException("null value"); // TODO
051            init(host, port, user, password);
052        }
053        public String getHost() {
054            return ( host );
055        }
056        public int getPort() {
057            return ( port );
058        }
059        public String getUser() {
060            return ( user );
061        }
062        public String getPassword() {
063            return ( password );
064        }
065        public boolean equals(final Object other) {
066            boolean same = false;
067            if (other instanceof LoginInfo) {
068                final LoginInfo that = (LoginInfo) other;
069                same = this.host.equals(that.host) &&
070                       this.port == that.port      &&
071                       this.user.equals(that.user) &&
072                       this.password.equals(that.password);
073            }
074            return ( same );
075        }
076        public int hashCode() {
077            return ( (int) 31 * host.hashCode() + 23 * port + 53 * user.hashCode() + 13 * password.hashCode() );
078        }
079        
080        private void init(final String host, final int port, final String user, final String password) {
081            this.host     = host;
082            this.port     = port;
083            this.user     = user;
084            this.password = password;
085        }
086        
087        public String toString() {
088            return ( host + port + user + password );
089        }
090    
091        public int compareTo(final LoginInfo that) {
092            final String thisKey = this.user + this.host + this.port;
093            final String thatKey = that.user + that.host + that.port;        
094            return ( thisKey.compareTo(thatKey) );
095        }
096    }
