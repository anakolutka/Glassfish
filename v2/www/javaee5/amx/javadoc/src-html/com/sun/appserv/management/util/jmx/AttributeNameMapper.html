030    package com.sun.appserv.management.util.jmx;
031    
032    
033    import java.util.Set;
034    
035    /**
036            Maps names to a derived name.  An example use for this might be to map a JMX
037            Attribute "foo-bar" to a suiteable "FooBar" form so that it can be used
038            as a Java identifier.
039     */
040    public interface AttributeNameMapper
041    {
042            /**
043                    Determine if the attribute name requires mapping.
044                    
045                    @param originalName     the original name from which a derived name is made
046             */
047            //public boolean                requiresMapping( String originalName );
048            
049            /**
050                    @param originalName the original/source/real name of the Attribute
051                    @param derivedName the name by which it should be known
052             */
053            public void                     addMapping( final String originalName, final String derivedName );
054            
055            /**
056                    Don't perform any mapping on this name.  Equivalent to calling
057                    addMapping( originalName, originalName )
058                    
059                    @param originalName the "real" name of the Attribute
060             */
061            public void                     dontMap( final String originalName );
062            
063            /**
064                    Setup mapping for all specified Attribute names.  These add to (or replace)
065                    any existing mappings.
066                    
067                    @param originalNames    all names from which should be derived names
068             */
069            public void                     deriveAll( final String[] originalNames );
070            
071            /**
072                Attempt to match the derived name to one of the candidates.
073                This facility is used when different runtime conditions
074                present different original names which must be mapped to the
075                same derived name.
076                <p>
077                If a name is matched it is added as a mapping and the
078                original name is returned.
079             */
080                public String
081            matchName(
082                final String   derivedName,
083                final String[] candidates );
084                
085            /**
086                    Maps Attribute names to another name.
087                    <p>
088                    A common use is to construct legal Java identifiers, so that they can
089                    be exposed in an MBean proxy with get/set routines.
090                    <p>
091                    For example "classpath-prefix" is not legal in a Java API; it could not
092                    generate the methods getclasspath-prefix() and setclasspath-prefix().
093                    <p>
094                    Any legal mapping is OK.  Suggested  possible mappings include:
095                    ClasspathPrefix, classpathPrefix, classpath_prefix, etc. These would
096                    result in the method names: getClasspathPrefix(), getclasspathPrefix(),
097                    getclasspath_prefix(), etc.
098                    
099                    @param originalName      original name
100             */
101            public String           originalToDerived( String originalName );
102            
103            /**
104                    Reverse the effect of originalToDerived()
105                    
106                    @param derivedName      name derived from the original one
107             */
108            public String           derivedToOriginal( String derivedName );
109            
110            
111            /**
112                    Get the entire set of Attribute names, consisting of the names
113                    that were derived, and the names that do not require mapping. 
114             */
115            public Set<String>                getAttributeNames( );
116    }
