033    /**
034            Identifies a Java VM being utilized by a server.
035     */
036    public interface JVM extends J2EEManagedObject
037    {
038            public final static String      J2EE_TYPE       = J2EETypes.JVM;
039            
040            /**
041                    Note that the Attribute name is case-sensitive
042                    "javaVendor" as defined by JSR 77.
043             */
044            public String           getjavaVendor();
045            
046            /**
047                    Note that the Attribute name is case-sensitive
048                    "javaVersion" as defined by JSR 77.
049             */
050            public String           getjavaVersion();
051            
052            /**
053                    Note that the Attribute name is case-sensitive
054                    "node" as defined by JSR 77.
055             
056                    @return the fully-qualified hostname
057             */
058            public String   getnode();
059    
060    
061    }
