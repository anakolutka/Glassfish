023    package com.sun.appserv.management.config;
024    
025    import com.sun.appserv.management.base.XTypes;
026    
027    
028    /**
029            Configuration for custom MBean.
030            <p>
031            When a Custom MBean is loaded, the object name specified via
032            the 'ObjectNameInConfig' Attribute is used.  The JMX Domain
033            will always be {@link #JMX_DOMAIN}.
034            <p>
035            If there is a name specified in  'ObjectNameInConfig' eg
036            "name=name1", that name is used and the name as returned
037            by {@link #getName} is not used within the ObjectName.
038            
039        <b>Questions</b>
040        <ul>
041        <li>
042            When setEnabled( true/false ) is called, are custom MBeans dynamically
043            loaded and unloaded?  Document this here.
044        </li>
045        <li>
046            How to obtain the ObjectName for the runtime MBean that
047            this config specifies?
048        </li>
049        <li>
050            Are runtime MBeans loaded only in the DAS, or in each server?
051            What about the Node Agent?  If they are loaded per server, 
052            what is put into the ObjectName to distinguish them?
053        </li>
054        <li>
055            All these either/or cases are confusing.  We should restrict
056            the object-name Attribute to properties, prohibit a 'name'
057            property, and require a 'type' property--my opinion--Lloyd
058        </li>
059        </ul>
060            
061            @see com.sun.appserv.management.config.DomainConfig#getCustomMBeanConfigMap
062     */
063    public interface CustomMBeanConfig
064        extends NamedConfigElement, Enabled, PropertiesAccess, ObjectType,
065        DeployedItemRefConfigReferent, Description
066    {
067    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
068            public static final String      J2EE_TYPE       = XTypes.CUSTOM_MBEAN_CONFIG;
069            
070            /**
071                The JMX domain in which all custom MBeans are registered by default.
072             */
073            public static final String  JMX_DOMAIN   = "user";
074            
075            /**
076                Get the ObjectName as configured (which could differ from the ObjectName
077                with which the MBean is registered--TBD).
078                This Attribute may not be changed; it is read-only.
079             */
080            public String getObjectNameInConfig();
081            
082            /**
083                Get the implementation class.
084                This Attribute is read-only.
085             */
086            public String getImplClassname();
087    }
