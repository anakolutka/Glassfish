042                    public static <T> T
043            getSingleton( final Set<T> s )
044            {
045                    if ( s.size() != 1 )
046                    {
047                            throw new IllegalArgumentException( s.toString() );
048                    }
049                    return( s.iterator().next() );
050            }
051            
052            /**
053                    Add all items in an array to a set.
054             */
055                    public static <T> void
056            addArray(
057                    final Set<T>      set,
058                    final T[]           array )
059            {
060                    for( int i = 0; i < array.length; ++i )
061                    {
062                            set.add( array[ i ] );
063                    }
064            }
065            
066            /**
067                    Convert a Set to a String[]
068             */
069                    public static String[]
070            toStringArray( final Set        set )
071            {
072                    final String[]  strings = new String[ set.size() ];
073                    
074                    final Iterator  iter    = set.iterator();
075                    int     i = 0;
076                    while ( iter.hasNext() )
077                    {
078                            strings[ i ]    = iter.next().toString();
079                            ++i;
080                    }
081                    
082                    return( strings );
083            }
084            
085            /**
086                    Convert a Set to a String[]
087             */
088                    public static Object[]
089            toArray( final Set<?>     set )
090            {
091                    final Object[]  names   = new Object[ set.size() ];
092                    set.toArray( names );
093                    
094                    return( names );
095            }

191            /**
192                    Create a new Set consisting of the contents of two sets.
193             */
194                    public static <T> Set<T>
195            newSet(
196                    final Set<? extends T>    s1,
197                    final Set<? extends T>    s2)
198            {
199                final Set<T>    result  = new HashSet<T>();
200                result.addAll( s1 );
201                result.addAll( s2 );
202                
203                    return result;
204            }
205            
206            /**
207                    Create a new Set consisting of the contents of three sets.
208             */
209                    public static <T> Set<T>
210            newSet(
211                    final Set<? extends T>    s1,
212                    final Set<? extends T>    s2,
213                    final Set<? extends T>    s3 )
214            {
215                return newSet( newSet( s1, s2 ), s3 );
216            }
217            
218            /**
219                    Create a new Set consisting of the contents of four sets.

223                    final Set<? extends T>    s1,
224                    final Set<? extends T>    s2,
225                    final Set<? extends T>    s3,
226                    final Set<? extends T>    s4 )
227            {
228                return newSet( newSet( s1, s2 ), newSet( s3, s4) );

247            /**
248                    Create a new Set containing all array elements.
249             */
250                    public static <T> Set<T>
251            newSet(
252                    final T []  objects,
253                    final int       startIndex,
254                    final int       numItems )
255            {
256                    final Set<T>      set     = new HashSet<T>();
257                    
258                    for( int i = 0; i < numItems; ++i )
259                    {
260                            set.add( objects[ startIndex + i ] );
261                    }
262    
263                    return( set );
264            }

266            /**
267                    Return a new Set of all items in both set1 and set2.
268             */
269                    public static <T> Set<T>
270            intersectSets(
271                    final Set<T>      set1,
272                    final Set<T> set2 )
273            {
274                    final Set<T>      result  = SetUtil.newSet( set1 );
275                    result.retainAll( set2 );
276                    
277                    return( result );
278            }
279            
280            /**
281                    Return a new Set of all items in set1 not in set2.
282             */
283                    public static <T> Set<T>
284            removeSet(
285                    final Set<T>       set1,
286                    final Set<T>   set2 )
287            {
288                    final Set<T>      result  = SetUtil.newSet( set1 );
289                    result.removeAll( set2 );
290                    
291                    return( result );
292            }
293            
294            /**
295                    Return a new Set of all items not common to both sets.
296             */
297                    public static <T> Set<T>
298            newNotCommonSet(
299                    final Set<T> set1,
300                    final Set<T> set2 )
301            {
302                    final Set<T>      result  = SetUtil.newSet( set1, set2 );
303                    final Set<T>      common  = intersectSets( set1, set2);
304                    
305                    result.removeAll( common );
306                    
307                    return( result );
308            }
