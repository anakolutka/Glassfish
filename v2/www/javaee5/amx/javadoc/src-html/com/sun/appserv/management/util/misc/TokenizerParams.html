030    package com.sun.appserv.management.util.misc;
031    
032    
033    /**
034            Parameters that TokenizerImpl accepts.
035     */
036    public class TokenizerParams
037    {
038            /**
039                    Delimiters are characters that separate tokens.
040             */
041            public String           mDelimiters;
042            
043            /**
044                    When multiple delimiters abut each other, are they all treated as
045                    a single delimiter, or as multiples with implied empty tokens
046                    between them?
047             */
048            public boolean          mMultipleDelimsCountAsOne;
049            
050            /**
051                    The escape char, allowed to be anything, but typically should
052                    be BACKSLASH.
053             */
054            public char                     mEscapeChar;
055            
056            /**
057                    Characters which may be escaped over and above the standard ones.
058             */
059            public String           mEscapableChars;
060            
061            /**
062                    When an invalid escape sequence is encountered, either an exception
063                    may be thrown, or the sequence may be emitted literally.
064             */
065            public boolean          mEmitInvalidEscapeSequencesLiterally;
066            
067            public final static char        BACKSLASH       = '\\';
068            public final static char        COMMA   = ',';
069            public final static char        DEFAULT_ESCAPE_CHAR     = BACKSLASH;
070            public final static String      DEFAULT_DELIMITERS      = "" + COMMA;
071            
072                    public
073            TokenizerParams()
074            {
075                    mDelimiters                                     = DEFAULT_DELIMITERS;
076                    mMultipleDelimsCountAsOne       = true;
077                    mEscapeChar                                     = DEFAULT_ESCAPE_CHAR;
078                    mEscapableChars                         = "" + DEFAULT_ESCAPE_CHAR;
079                    mEmitInvalidEscapeSequencesLiterally    = false;
080            }
081            
082                    public void
083            ensureDelimitersEscapable()
084            {
085                    for( int i = 0; i < mDelimiters.length(); ++i )
086                    {
087                            final char      delim   = mDelimiters.charAt( i );
088                            
089                            if ( mEscapableChars.indexOf( delim ) < 0 )
090                            {
091                                    mEscapableChars = mEscapableChars + delim;
092                            }
093                    }
094            }
095    }
