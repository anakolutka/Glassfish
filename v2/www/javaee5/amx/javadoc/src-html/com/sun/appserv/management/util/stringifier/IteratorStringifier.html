030    package com.sun.appserv.management.util.stringifier;
031    
032    import java.util.Iterator;
033    
034    
035    /**
036            Stringifies an Iterator, using an optional element Stringifier
037     */
038     
039    public final class IteratorStringifier extends IteratorStringifierBase
040    {
041                    public 
042            IteratorStringifier()
043            {
044                    super();
045            }
046            
047                    public 
048            IteratorStringifier( String delim )
049            {
050                    super( delim );
051            }
052            
053                    public 
054            IteratorStringifier( Stringifier elementStringifier )
055            {
056                    super( elementStringifier );
057            }
058            
059                    public 
060            IteratorStringifier( String delim, Stringifier elementStringifier )
061            {
062                    super( delim, elementStringifier );
063            }
064            
065            
066                    public void
067            stringifyElement(
068                    Object                  elem,
069                    String                  delim,
070                    StringBuffer    buf)
071            {
072                    if ( elem == null )
073                    {
074                            buf.append( "null" );
075                    }
076                    else
077                    {
078                            buf.append( mElementStringifier.stringify( elem ) );
079                    }
080            }
081            
082            
083            /*
084                    Static variant when direct call will suffice.
085             */
086                    public static String
087            stringify( Iterator iter, String delim )
088            {
089                    final IteratorStringifier       stringifier     = new IteratorStringifier( delim );
090                    
091                    return( stringifier.stringify( iter, delim, SmartStringifier.DEFAULT ) );
092            }
093            
094            
095            public final static IteratorStringifier DEFAULT = new IteratorStringifier( "," );
096    }
