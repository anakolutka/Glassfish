023    package com.sun.appserv.management.ext.wsmgmt;
024    
025    import com.sun.appserv.management.base.MapCapable;
026    
027    /**
028     * Provides details of a SOAP invocation. 
029     *
030     * @since AppServer 9.0
031     */
032    public interface MessageTrace extends MapCapable
033     {
034    
035        /** This interface's class name */
036        public final static String  CLASS_NAME    =
037                    "com.sun.appserv.management.ext.wsmgmt.MessageTrace";
038    
039        /**
040         * Returns the total size in bytes of the request content. This is 
041         * SOAP content size including body and headers. 
042         *
043         * @return  SOAP request content size in bytes
044         */
045        public int getRequestSize();
046    
047        /**
048         * Returns the total size in bytes of the response content. This is 
049         * SOAP content size including body and headers. 
050         *
051         * @return  SOAP response content size in bytes
052         */
053        public int getResponseSize();
054    
055        /**
056         * Returns the SOAP request content. This contains SOAP body and headers.
057         *
058         * @return  SOAP request content
059         */
060        public String getRequestContent();
061    
062        /**
063         * Returns the SOAP response content. This contains SOAP body and headers. 
064         * 
065         * @return SOAP response content
066         */
067        public String getResponseContent();
068    
069        /**
070         * Returns the transport type. Known types are "HTTP", "JMS" and "SNMP".
071         *
072         * @return  transport type
073         */
074        public String getTransportType();
075    
076        /**
077         * Returns the comma separated list of HTTP request header names and its 
078         * values. Null if transport type is not HTTP. 
079         *
080         * @return  HTTP request headers
081         */
082        public String getHTTPRequestHeaders();
083    
084        /**
085         * Returns the comma separated list of HTTP response header names and its
086         * values. Null if transport type is not HTTP. 
087         *
088         * @return  HTTP response headers
089         */
090        public String getHTTPResponseHeaders();
091    
092        /**
093         * Returns the IP address or host name of the client machine. 
094         *
095         * @return  IP address or host name of client
096         */
097        public String getClientHost();
098    
099        /**
100         * Returns the caller principal name. 
101         *
102         * @return  caller principal name
103         */
104        public String getPrincipalName();
105    
106        /**
107         * Returns the response time in milli seconds for this web service 
108         * operation. 
109         * 
110         * @return  reponse time in milli seconds
111         */
112        public long getResponseTime();
113    
114        /**
115         * Returns the fault code. 
116         *
117         * @return  fault code for this web service or null
118         */
119        public String getFaultCode();
120    
121        /**
122         * Returns the fault string. 
123         *
124         * @return  fault string for this web service or null
125         */
126        public String getFaultString();
127    
128        /**
129         * Returns the fault actor. 
130         *
131         * @return  fault actor for this web service
132         */
133        public String getFaultActor();
134    
135        /**
136         * Returns the message id.
137         *
138         * @return  message id
139         */
140        public String getMessageID();
141    
142        /**
143         * Returns the name of the application.
144         *
145         * @return  application name
146         */
147        public String getApplicationID();
148    
149        /**
150         * Returns the name of the webservice endpoint.
151         *
152         * @return  name of the webservice endpoint
153         */
154        public String getEndpointName();
155    
156        /**
157         * Returns the time stamp when the SOAP message was received.
158         *
159         * @return the time stamp when the SOAP message was received
160         */
161        public long getTimestamp();
162    
163        /**
164         * Returns if during this web service invocation call flow was enabled or
165         * not
166         *
167         * @return true, if during this web service invocation call flow was
168         * enabled otherwise flase.
169         */
170        public boolean isCallFlowEnabled();
171    }
