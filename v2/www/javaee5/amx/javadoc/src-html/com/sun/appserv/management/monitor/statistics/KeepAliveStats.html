030    package com.sun.appserv.management.monitor.statistics;
031    
032    import javax.management.j2ee.statistics.Stats;
033    import javax.management.j2ee.statistics.CountStatistic;
034    
035    /**
036            @see com.sun.appserv.management.monitor.KeepAliveMonitor
037     */
038    public interface KeepAliveStats extends Stats
039    {
040        /** 
041         * @return number of keep-alive connections that timed out.
042         */    
043            public CountStatistic   getCountTimeouts();
044            
045        /** 
046            @return number of connections in keep-alive mode
047         */    
048            public CountStatistic   getCountConnections();
049            
050        /** 
051            Gets the number of seconds before a keep-alive connection that has
052            been idle times out and is closed.
053         
054            @return Keep-alive timeout in number of seconds
055         */    
056            public CountStatistic   getSecondsTimeouts();
057            
058        /** 
059            @return maximum number of concurrent connections in keep-alive mode
060         */    
061            public CountStatistic   getMaxConnections();
062            
063    
064        /** 
065            @return number of keep-alive connections that were rejected.
066         */    
067            public CountStatistic   getCountRefusals();
068            
069        /** 
070            @return Number of keep-alive connections that were closed
071         */    
072            public CountStatistic   getCountFlushes();
073            
074        /** 
075            @return Number of requests received by connections in keep-alive mode.
076         */    
077            public CountStatistic   getCountHits();
078    }
