025    package com.sun.appserv.management.config;
026    
027    import com.sun.appserv.management.base.XTypes;
028    
029    
030    
031    /**
032        Mixin interface for access to library path.
033     */
034    public interface Libraries
035    {
036            /**
037                Optional Attribute (may be null).
038                <p>                                            
039            These paths could be either relative [relative to                               
040            {com.sun.aas.instanceRoot}/lib/applibs] or absolute paths.    
041            These dependencies appears *after* the libraries defined in   
042            classpath-prefix in the java-config and *before* the          
043            application server provided over-rideable jar set. The        
044            libraries would be made available to the application in the   
045            order in which they were specified.
046            
047            @since AppServer 9.0   
048             */
049            public String[] getLibraries();
050            
051            /**
052                Replaces the existing libraries Attribute. Certain
053                system applications may not allow changing this Attribute
054                (read only).
055                                                        
056            @see #getLibraries
057            @since AppServer 9.0   
058             */
059            public void setLibraries( String[] libraries );
060    }
