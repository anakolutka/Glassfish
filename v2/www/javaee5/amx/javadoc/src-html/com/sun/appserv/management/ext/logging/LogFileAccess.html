025    package com.sun.appserv.management.ext.logging;
026    
027    
028    /**
029            Provides access to log files.
030            
031            @since AS 9.0
032     */
033    public interface LogFileAccess
034    {
035        /**
036            Value meaning the most current version of the log file.
037         */
038        public static final String  MOST_RECENT_NAME = "MostRecentLogFileName";
039         
040            /**
041                    Key designating the server.log log files.
042                    <i>Not necessarily the same as the file name of the log file</i>.
043             */
044        public static final String  SERVER_KEY      = "server";
045        
046            /**
047                    Key designating the access.log log file.
048                    <i>Not necessarily the same as the file name of the log file</i>.
049             */
050        public static final String  ACCESS_KEY      = "access";
051        
052        /**
053            Keys which may be used to specify which log file to access.
054            Legal values include:
055            <ul>
056            <li>{@link #SERVER_KEY}</li>
057            <li>{@link #ACCESS_KEY} <b>is not supported</b></li>
058            </ul>
059            @return a String[] of the legal keys which designate log files
060         */
061        public String[]     getLogFileKeys();
062        
063        
064            /**
065                    The names returned are <i>not</i> full paths but the simple file
066                    names of the log files.
067                    The last name in the resulting String[] will always be that of the
068                    current log file.  In other words if the String[] las length 3,
069                    then result[2] will be the name of the current log file.
070                    <p>
071                    Note that it is possible for log file rotation to occur after making
072                    this call, thus rendering the list incomplete.
073                    <p>
074                    The resulting names may be passed to {@link #getLogFile} to retrieve
075                    the contents.
076                    <p>
077                    The only legal key supported is {@link #SERVER_KEY}.
078                    
079                    @param key      a key specifying the type of log file
080                    @return String[] of all log filenames 
081             */
082            public String[] getLogFileNames( final String key );
083            
084            /**
085                    The entire specified log file is read, converted into a String, and returned.
086                    The resulting String may be quite large.
087                    <p>
088                    The only legal key supported is {@link #SERVER_KEY}.
089                    
090                    @param key a legal key designating a log file
091                    @param fileName a log file name as returned by {@link #getLogFileNames} or
092                    {@link #MOST_RECENT_NAME} if current log file is desired.
093                    @return the contents of the specified log file in a String
094                    @see #getLogFileKeys
095             */
096            public String getLogFile( final String key, final String fileName );
097            
098        /**
099                    Rotate all log files as soon as possible.  May return
100                    prior to the rotation occuring.
101         */
102        public void rotateAllLogFiles();
103        
104        
105        /**
106            Rotate the log file of the specified type.  Legal values are those
107            returned by {@link #getLogFileKeys}.
108            Legal values include:
109            <ul>
110            <li>{@link #SERVER_KEY}</li>
111            <li>{@link #ACCESS_KEY} <b>is not supported</b></li>
112            </ul>
113            @param key
114         */
115        public void rotateLogFile( String key );
116    }
