029    package com.sun.appserv.management.util.jmx;
030    
031    import java.util.Set;
032    
033    import javax.management.ObjectName;
034    import javax.management.MBeanServerConnection;
035    import javax.management.MalformedObjectNameException;
036    
037    public interface ObjectNameQuery
038    {
039            /**
040                    Return the ObjectNames of all MBeans whose properties match all the specified
041                    regular expressions.  Both property names and values may be searched.
042                    
043                    A starting set may be specified by using an ObjectName pattern.
044                    This can greatly improve the performance of the search by restricting the
045                    set of MBeans which are examined; otherwise all registered MBeans must be examined.
046                    
047                    The regexNames[ i ] pattern corresponds to regexValues[ i ].  A value of null
048                    for any item is taken to mean "match anything".  Thus specifing null for 
049                    'regexNames' means "match any name" and specifying regexNames[ i ] = null means
050                    to match only based on regexValues[ i ] (and vice versa).
051                    
052                    @param startingSet                      optional ObjectName pattern for starting set to search
053                    @param regexNames                       optional series of regular expressions for Property names
054                    @param regexValues                      optional series of regular expressions for Property values
055                    @return                                         array of ObjectName (may be of zero length)
056             */
057            Set<ObjectName>   matchAll( Set<ObjectName> startingSet, String [] regexNames, String [] regexValues );
058                                    
059            /**
060                    Return the ObjectNames of all MBeans whose properties match any of the specified
061                    regular expressions.  Both property names and values may be searched.
062                    
063                    A starting set may be specified by using an ObjectName pattern.
064                    This can greatly improve the performance of the search by restricting the
065                    set of MBeans which are examined; otherwise all registered MBeans must be examined.
066                    
067                    
068                    The regexNames[ i ] pattern corresponds to regexValues[ i ].  A value of null
069                    for any item is taken to mean "match anything".  Thus specifing null for 
070                    'regexNames' means "match any name" and specifying regexNames[ i ] = null means
071                    to match only based on regexValues[ i ] (and vice versa).
072                    
073                    @param startingSet                      optional ObjectName pattern for starting set to search
074                    @param regexNames                       optional series of regular expressions for Property names
075                    @param regexValues                      optional series of regular expressions for Property values
076                    @return                                         array of ObjectName (may be of zero length)
077             */
078            Set<ObjectName>   matchAny( Set<ObjectName> startingSet, String [] regexNames, String [] regexValues );
079    }
