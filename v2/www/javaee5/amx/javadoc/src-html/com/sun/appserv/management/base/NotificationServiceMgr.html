023    package com.sun.appserv.management.base;
024    
025    import java.util.Map;
026    
027    import javax.management.InstanceNotFoundException;
028    
029    import com.sun.appserv.management.base.Container;
030    
031    /**
032            Manages the creation and removal of NotificationService MBeans.
033     */
034    public interface NotificationServiceMgr extends AMX, Utility, Container
035    {
036    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
037            public static final String      J2EE_TYPE                       = XTypes.NOTIFICATION_SERVICE_MGR;
038            
039            /**
040                    @return Map of all {@link NotificationService} keyed by name
041             */     
042            public Map<String,NotificationService> getNotificationServiceMap( );
043            
044            /**
045                Get the specified {@link NotificationService}.
046                @param name
047             */
048            public NotificationService              getNotificationService( final String name );
049            
050            /**
051                    Create a new NotificationService (j2eeType={@link XTypes#NOTIFICATION_SERVICE}).
052                    {@link NotificationService#getName} can later be used
053                    to retrieve the NotificationService via {@link #getNotificationService}.
054                    
055                    @param userData
056                    @param bufferSize
057                    @return NotificationService
058             */
059            public NotificationService              createNotificationService( Object userData, int bufferSize );
060            
061            /**
062                Remove the specified {@link NotificationService}.
063                @param name
064             */
065            public void removeNotificationService( final String name )
066                                             throws InstanceNotFoundException;
067    }
