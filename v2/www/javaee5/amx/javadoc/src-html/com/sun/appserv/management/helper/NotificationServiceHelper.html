029    package com.sun.appserv.management.helper;
030    
031    import java.util.Map;
032    
033    import javax.management.Notification;
034    
035    import com.sun.appserv.management.base.NotificationService;
036    
037    /**
038            Helps make it easier to get Notifications from a NotificationService
039     */
040    public final class NotificationServiceHelper
041    {
042            final NotificationService       mProxy;
043            final Object                                    mBufferID;
044            long                                                    mNextSequenceNumber;
045            
046                    public
047            NotificationServiceHelper(
048                    final NotificationService       proxy,
049                    final Object                                    bufferID )
050            {
051                    mProxy                          = proxy;
052                    mBufferID                       = bufferID;
053                    
054                    mNextSequenceNumber     = 0;
055            }
056            
057            
058                    public NotificationService
059            get()
060            {
061                    return( mProxy );
062            }
063            
064            /**
065                    Get any outstanding Notifications and return them.
066             */
067                    public Notification[]
068            getNotifications()
069            {
070                    final Map<String,Object>  result  =
071                            mProxy.getBufferNotifications( mBufferID, mNextSequenceNumber);
072                    
073                    final Long      nextSequenceNumber      = (Long)
074                            result.get( mProxy.NEXT_SEQUENCE_NUMBER_KEY );
075                    mNextSequenceNumber     = nextSequenceNumber.longValue();
076                    
077                    final Notification[]    notifs  =
078                            (Notification[])result.get( mProxy.NOTIFICATIONS_KEY );
079                    
080                    return( notifs );
081            }
082    }
