023    package com.sun.appserv.management.base;
024    
025    import java.util.Set;
026    import java.util.Map;
027    
028    /**
029            All MBeans which <i>logically contain</i> other MBeans implement this interface;
030            such an MBean is considered a Container, and the contained MBean is considered
031            a Containee.  The interface indicates the <i>potential</i> to contain other
032            MBeans; at any given time MBeans may or may not be contained.
033     */
034    public interface Container extends AMX
035    {
036            /**
037                    Attribute returned by getContaineeJ2EETypes().
038             */
039            public static final String      ATTR_CONTAINEE_J2EE_TYPES       = "ContaineeJ2EETypes";
040            
041            /**
042                    @return Set of String of all <i>possible</i> j2eeTypes contained within this item
043                    @see com.sun.appserv.management.base.Util#getNamesSet
044             */
045            public Set<String>                getContaineeJ2EETypes();
046            
047            /**
048                    Return a Map keyed by j2eeType. The <i>value</i> corresponding to each
049                    key (j2eeType) is another Map, as returned from {@link #getContaineeMap}.
050                    <p>
051                    If the passed Set is null, then all types are obtained.  Pass
052                    the set returned from {@link #getContaineeJ2EETypes} to get all currently
053                    present containees.
054                    
055                    @param j2eeTypes        the j2eeTypes to look for, or null for all
056                    @return Map (possibly empty) of <i>j2eeType</i>=&lt;Map of <i>name</i>=<i>AMX</i>&gt;
057             */
058            public <T extends AMX> Map<String,Map<String,T>>              getMultiContaineeMap( final Set<String> j2eeTypes );
059            
060            /**
061                    Each key in the resulting Map is a String which is the value of the 
062                    AMX.NAME_KEY for that {@link AMX}, which is the value.
063                    
064                    @param j2eeType the j2eeType to look for
065                    @return Map <i>name</i>=<i>AMX</i>
066             */
067            public <T extends AMX> Map<String,T>                getContaineeMap( final String j2eeType );
068            
069            /**
070                    Obtain the singleton MBean having the specified type.
071                    
072                    @param j2eeType
073                    @return {@link AMX} of specified j2eeType or null if not present
074                    @throws IllegalArgumentException if there is more than one item of this type
075                    @see Util#getNamesSet
076             */
077            public <T extends AMX> T  getContainee( final String j2eeType );
078            
079            /**
080                    @return all containees having the specified j2eeType.
081                    @see Util#getNamesSet
082             */
083            public <T extends AMX> Set<T>       getContaineeSet( final String j2eeType );
084            
085            /**
086                    Same as getContaineeSet( getContaineeJ2EETypes() )
087                    @return all containees of any j2eeType
088                    @see Util#getNamesSet
089                    @see #getContaineeSet(java.util.Set)
090             */
091            public <T extends AMX> Set<T>       getContaineeSet( );
092            
093            /**
094                    @return all containees having the specified j2eeType(s).
095                    @see Util#getNamesSet
096                    @see #getMultiContaineeMap
097             */
098            public <T extends AMX> Set<T>       getContaineeSet( final Set<String> j2eeTypes );
099            
100            
101            /**
102                    @return all containees having one of the specified types
103                    and the specified name.
104                    @see com.sun.appserv.management.base.Util#getNamesSet
105             */
106            public <T extends AMX> Set<T>       getByNameContaineeSet(
107                                            final Set<String> j2eeTypes, final String name );
108            
109            /**
110                    Get a singleton containee having the specified j2eeType and name.
111                    @param j2eeType the j2eeType of the contained 
112                    @param name             the name of the contained  (as found in "name" property)
113                    @return AMX or null if not found
114                    @see Util#getNamesSet
115            */
116            public <T extends AMX> T  getContainee( final String j2eeType, final String name );
117    }
