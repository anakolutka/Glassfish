006    package com.sun.appserv.management.monitor.statistics;
007    
008    import javax.management.j2ee.statistics.Stats;
009    import javax.management.j2ee.statistics.CountStatistic;
010    import javax.management.j2ee.statistics.TimeStatistic;
011    
012    /**
013     * Stats interface for the monitorable attributes of a 
014     * Load Balancer at the application level
015     */
016    public interface LoadBalancerApplicationStats extends Stats {
017        
018        /** Returns average response time in milliseconds
019         * @return an instance of {@link TimeStatistic}     
020         */
021        public TimeStatistic getAverageResponseTime();
022    
023        /** Returns the number of times service failed
024         * @return an instance of {@link CountStatistic}     
025         */
026        public CountStatistic getServiceErrorCount();
027        
028        /** Returns the percentage of time service failed to total service requests
029         * @return an instance of {@link CountStatistic}     
030         */
031        public CountStatistic getServiceErrorPercentage();
032        
033        /** Returns the percentage of uptime to total time
034         * @return an instance of {@link CountStatistic}     
035         */
036        public CountStatistic getServiceUptimePercentage();
037    
038        /** Returns the number of sessions
039         * @return an instance of {@link CountStatistic}     
040         */
041        public CountStatistic getSessionCount();
042    
043        /** Returns the number of times all idempotent urls in this application are 
044         * accessed
045         * @return an instance of {@link CountStatistic}     
046         */
047        public CountStatistic getIdempotentUrlAccessCount();
048        
049        /** Returns the percentage of number of times all idempotent urls in this 
050         * application are accessed to all urls accessed
051         * @return an instance of {@link CountStatistic}     
052         */
053        public CountStatistic getIdempotentUrlAccessPercentage();
054    }
