026    
027    import java.util.Collections;
028    import java.util.Map;
029    import java.util.Set;
030    import java.util.List;

032

036    /**
037            Provides summary information about important logging events.
038            <big>PRELIMINARY--SUBJECT TO CHANGES/ADDITIONS</big>
039            
040            @since AS 9.0
041     */
042    public interface LogAnalyzer
043    {
044            /**
045                    Key into any Map returned from {@link #getErrorInfo}.
046                    value is of type Long.
047            */
048            public static final String TIMESTAMP_KEY                = "TimeStamp";
049            
050            /**
051                    Key into any Map returned from {@link #getErrorInfo}.
052                    value is of type Long.
053            */
054            public static final String SEVERE_COUNT_KEY             = "SevereCount";
055            
056            /**
057                    Key into any Map returned from {@link #getErrorInfo}.
058                    value is of type Long.
059            */
060            public static final String WARNING_COUNT_KEY    = "WarningCount";
061            
062            /**
063                    Key into any Map returned from {@link #getErrorDistribution}.
064                    value is of type String.
065            */
066            public static final String MODULE_NAME_KEY      = "ModuleName";
067            
068            
069            /**
070                    Get a summary of the {@link Level#SEVERE} and {@link Level#WARNING} log
071                    entries for the known history. Each entry in the resulting array is a
072                    Map with the following keys:
073                    <ul>
074                    <li>{@link #TIMESTAMP_KEY} of type Long</li>
075                    <li>{@link #SEVERE_COUNT_KEY} of type Integer</li>
076                    <li>{@link #WARNING_COUNT_KEY} of type Integer</li>
077                    </ul>
078                    The entries are arranged from oldest to newest with the last entry being
079                    the most recent.
080                    <p>
081                    The timestamp obtained from each Map may be used as the timestamp when
082                    calling {@link #getErrorDistribution}. For example:<br>
083    <code>
084    final Map<String,Number>[]        infos   = logging.getErrorInfo();<br>
085    for( int i = 0; i < infos.length; ++i ) {<br>
086            final Map<String,Object>  info    = infos[ i ];<br>
087            final long timestamp    = ((Long)info.get( TIMESTAMP_KEY )).longValue();<br>
088            <br>
089            Map<String,Number>        counts  = getErrorDistribution( timestamp );<br>
090    }
091    </code>
092                    
093                    @return Map<String,Number>
094             */
095            public Map<String,Number>[]       getErrorInfo();
096            
097            
098            /**
099                    Get the number of log entries for a particular timestamp of a particular {@link Level}
100                    for all modules.  SEVERE and WARNING are the only levels supported.
101                    <p>
102                    The resulting Map is keyed by the module ID, which may be any of the values
103                    found in {@link LogModuleNames} or any valid Logger name.
104                    The corresponding value
105                    is the count for that module of the requested level.
106                    <p>
107                    
108                    @param timestamp a timestamp as obtained using TIME_STAMP_KEY from one of the Maps
109                    returned by {@link #getErrorInfo}.  Note that it is a 'long' not a 'Long' and is required.
110                    @param level
111                    @return Map<String,Integer>
112             */
113            public Map<String,Integer>        getErrorDistribution(long timestamp, String level);
114            
115            /**
116                @return all the logger names currently in use
117             */
118            public String[] getLoggerNames();
119            
120            /**
121                @return all the logger names currently in use under this logger.
122             */
123        public String[]   getLoggerNamesUnder( String loggerName );
124        
125        /**
126            Set the number of intervals error statistics should be maintained.
127         
128            @param numIntervals number of intervals
129         */
130        public void setKeepErrorStatisticsForIntervals(final int numIntervals );
131        
132        /*
133            See {@link #setErrorStatisticsIntervals}.
134         */
135        public int  getKeepErrorStatisticsForIntervals();
136    
137        /**
138            Set the duration of an interval.
139        
140            @param minutes The duration of an interval in minutes.
141         */
142        public void setErrorStatisticsIntervalMinutes(final long minutes);
143        
144        /*
145            See {@link #setErrorStatisticsIntervalMinutes}.
146         */
147        public long getErrorStatisticsIntervalMinutes(); 
148    }
