030    package com.sun.appserv.management.config;
031    
032    import java.util.Map;
033    import java.util.Date;
034    
035    import com.sun.appserv.management.base.XTypes;
036    import com.sun.appserv.management.base.Container;
037    
038    /**
039       Configuration for the lb-config element.
040       @see com.sun.appserv.management.ext.lb.LoadBalancer 
041     */
042    public interface LBConfig extends 
043        AMXConfig, PropertiesAccess, NamedConfigElement,
044            ServerRefConfigCR, ClusterRefConfigCR, Container {
045        
046        /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
047        public static final String  J2EE_TYPE = XTypes.LB_CONFIG;
048    
049        /**
050          Returns the period, in seconds, within which a server must return a response 
051          or otherwise it will be considered unhealthy. Must be greater than or 
052          equal to 0. 
053         */
054        public String getResponseTimeoutInSeconds();
055    
056        /**
057          Set the period, in seconds, within which a server must return a response or
058          otherwise it will be considered unhealthy. Default value is 60 seconds. 
059          Must be greater than or equal to 0. A value of 0 effectively turns off 
060          this check functionality, meaning the server will always be considered healthy.
061         */
062        public void setResponseTimeoutInSeconds(final String responseTimeoutInSeconds);
063    
064        /**
065          Returns a boolean flag indicating how load-balancer will route HTTPS
066          requests. If true, then an HTTPS request to the load-balancer will 
067          result in an HTTPS request to the server; if false, then HTTPS requests 
068          to the load-balancer result in HTTP requests to the server. 
069         */
070        public boolean getHttpsRouting();
071    
072        /**
073          Set the boolean flag indicating how load-balancer will route HTTPS
074          requests. If true, then an HTTPS request to the load-balancer will 
075          result in an HTTPS request to the server; if false, then HTTPS requests 
076          to the load-balancer result in HTTP requests to the server. 
077          Default is to use HTTP (i.e. value of false);
078         */
079        public void setHttpsRouting(final boolean value);
080        
081        /**
082          Returns the maximum period, in seconds, that a change to the load
083          balancer configuration file takes before it is detected by
084          the load balancer and the file reloaded. A value of 0
085          indicates that reloading is disabled. 
086         */
087        public String getReloadPollIntervalInSeconds();
088    
089        /**
090          Set the maximum period, in seconds, that a change to the load balancer 
091          configuration file takes before it is detected by the load balancer and 
092          the file reloaded. A value of 0 indicates that reloading is disabled. 
093          Default period is 1 minute (60 seconds)
094         */
095        public void setReloadPollIntervalInSeconds(final String reloadPollIntervalInSeconds);
096    
097        /**
098          Returns the boolean flag that determines whether monitoring is switched
099          on or not. Default is that monitoring is switched off (false)
100         */
101        public boolean getMonitoringEnabled();
102    
103        /**
104          Set the boolean flag that determines whether monitoring is switched
105          on or not. Default is that monitoring is switched off (false)
106         */
107        public void setMonitoringEnabled(final boolean value);
108       
109        /**
110          Returns the boolean flag that determines whether a route cookie is or is
111          not enabled.
112         */
113        public boolean getRouteCookieEnabled();
114    
115        /**
116          Set the boolean flag that determines whether a route cookie is or is
117          not enabled. Default is enabled (true).
118         */
119        public void setRouteCookieEnabled(final boolean value);
120        
121        /**
122          Calls Container.getContaineeMap(XTypes.CLUSTER_REF_CONFIG ).
123          @return Map of ClusterRefConfig MBean proxies, keyed by name.
124          @see com.sun.appserv.management.base.Container#getContaineeMap
125         */
126        public Map<String,ClusterRefConfig> getClusterRefConfigMap();
127    
128        /**
129          Calls Container.getContaineeMap(XTypes.SERVER_REF_CONFIG ).
130          @return Map of ServerRefConfig MBean proxies, keyed by name.
131          @see com.sun.appserv.management.base.Container#getContaineeMap
132         */
133        public Map<String,ServerRefConfig> getServerRefConfigMap();    
134    }
