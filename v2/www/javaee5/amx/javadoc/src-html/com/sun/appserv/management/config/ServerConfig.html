031    package com.sun.appserv.management.config;
032    
033    
034    
035    import java.util.Map;
036    
037    import com.sun.appserv.management.config.TemplateResolver;
038    import com.sun.appserv.management.base.Container;
039    
040    /**
041            Base interface for server configuration for the &lt;server&gt; element.
042            Does not provide ability to access resource or application-ref; see
043            {@link StandaloneServerConfig} and {@link ClusteredServerConfig}.
044     */
045    public interface ServerConfig
046            extends PropertiesAccess, SystemPropertiesAccess,
047            NamedConfigElement, TemplateResolver, Container
048    {
049            /**
050                    Get the name of the config element referenced by this server.
051             */
052            public String           getReferencedConfigName( );
053            
054            /**
055                    Get the name of the node-agent element referenced by this server.
056             */
057            public String           getReferencedNodeAgentName( );
058            
059            /**
060                    Calls Container.getContaineeMap( XTypes.DEPLOYED_ITEM_REF_CONFIG ).
061                    @return Map of all DeployedItemRefConfig MBean proxies, keyed by name.
062                    @see com.sun.appserv.management.base.Container#getContaineeMap
063             */
064            public Map<String,DeployedItemRefConfig>  getDeployedItemRefConfigMap();
065            
066            /**
067                    Calls Container.getContaineeMap( XTypes.RESOURCE_REF_CONFIG ).
068                    @return Map of all ResourceRefConfig MBean proxies, keyed by name.
069                    @see com.sun.appserv.management.base.Container#getContaineeMap
070             */
071            public Map<String,ResourceRefConfig>      getResourceRefConfigMap();
072    
073       
074            /**
075                <b>EE only</b>
076                Return the load balancer weight for this server.
077                This is used by both IIOP and HTTP load balancer. Default value is 1.
078                @since AppServer 9.0
079          */
080        public String getLBWeight();
081            
082            /**
083                <b>EE only</b>
084                Set the load balancer weight for this server to the specified value.
085                @since AppServer 9.0
086            */
087            public void setLBWeight(final String weight);
088    }
