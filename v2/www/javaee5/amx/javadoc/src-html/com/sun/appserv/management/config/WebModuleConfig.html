030    package com.sun.appserv.management.config;
031    
032    import java.util.Map;
033    
034    import com.sun.appserv.management.base.XTypes;
035    import com.sun.appserv.management.base.Container;
036    
037    /**
038             Configuration for the &lt;web-module&gt; element.
039    */
040    
041    public interface WebModuleConfig
042        extends ModuleConfig, Libraries, Container, WebServiceEndpointConfigCR
043    {
044    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
045            public static final String      J2EE_TYPE       = XTypes.WEB_MODULE_CONFIG;
046            
047            /**
048            The ContextRoot must match the pattern for the hpath production  
049            in RFC 1738 which can be found at:                            
050            http://www.w3.org/Addressing/rfc1738.txt. This is flattened   
051            to the following regular expression in XML Schema's pattern   
052            language:
053    
054            <pre>                                     
055            ([a-zA-Z0-9$\-_.+!*'(),]|%[0-9A-Fa-f][0-9A-Fa-f]|;|:|&|=)*(/([
056            -zA-Z0-9$\-_.+!*'(),]|%[0-9A-Fa-f][0-9A-Fa-f]|;|:|&|=)*)*
057            </pre>
058              
059            Note that this includes the null or empty context root and    
060            permits but does not require a context root to start with the 
061            '/' character (including a context root which is simply the   
062            '/' character).
063            @see #setContextRoot                                       
064            */
065          
066            public String   getContextRoot();
067            
068            /**
069            @see #getContextRoot                                       
070            */
071            public void             setContextRoot( String value );
072    
073            /**
074                    Controls whether availability is enabled    
075                    for HTTP Session Persistence. If this is "false", then all    
076                    session persistence is disabled for the given web module. If  
077                    it is "true" (and providing that all the availability-enabled 
078                    attributes above in precedence are also "true", then the web  
079                    module may be ha enabled. Finer-grained control exists at     
080                    lower level (see sun-web.xml).                               
081    
082                    @see #setAvailabilityEnabled
083             */
084            public boolean  getAvailabilityEnabled();
085    
086            /**
087                    @see #getAvailabilityEnabled
088            */       
089            public void             setAvailabilityEnabled( boolean enabled );
090            
091        public Map<String,WebServiceEndpointConfig> getWebServiceEndpointConfigMap();
092    }
