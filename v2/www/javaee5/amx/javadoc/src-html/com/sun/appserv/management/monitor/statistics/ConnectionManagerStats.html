023    package com.sun.appserv.management.monitor.statistics;
024    
025    import javax.management.j2ee.statistics.Stats;
026    import javax.management.j2ee.statistics.CountStatistic;
027    import javax.management.j2ee.statistics.BoundedRangeStatistic;
028    
029    /**
030            @see com.sun.appserv.management.monitor.ConnectionManagerMonitor
031     */
032    public interface ConnectionManagerStats extends Stats
033    {
034        /**
035                @return total number of connections to the ORB.   
036             */
037            public BoundedRangeStatistic getTotalConnections();
038            
039        
040        /** 
041                @return     total number of idle connections to the ORB    
042             */
043            public CountStatistic           getConnectionsInUse();
044            
045        /**
046                @return             total number of in-use connections to the ORB
047             */
048            public CountStatistic           getConnectionsIdle();
049    }
