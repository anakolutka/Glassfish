005    package amxsamples;

007    import com.sun.appserv.management.DomainRoot;
008    
009    
010    /**
011        Write your own code, then run it in the command line eg:
012        <p>
013        <code>Enter Command > mystuff.MyCmd [args]</code>
014        
015        Implement the runCmd() method any way you like.
016     */
017    public class CmdTemplate
018    {
019                    public 
020            CmdTemplate()
021            {
022            }
023            
024                protected String
025            argsToString(final String[] args)
026            {
027                final StringBuilder builder = new StringBuilder();
028                
029                final String NEWLINE    = System.getProperty( "line.separator" );
030                
031                builder.append( this.getClass().getName() + ".runCmd()" );
032                builder.append( NEWLINE );
033                for( int i = 0; i < args.length; ++i )
034                {
035                    builder.append( "arg[" + i + "] = "  + args[i] );
036                    builder.append( NEWLINE );
037                }
038                
039                return builder.toString();
040            }
041            
042            /**
043                args[0] is the class name
044             */
045                public Object
046            runCmd(
047                final DomainRoot    domainRoot,
048                final String[]      args )
049            {
050                return argsToString( args );
051            }
052    }
