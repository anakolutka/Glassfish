030    package com.sun.appserv.management.j2ee;
031    
032    import java.util.Map;
033    
034    import com.sun.appserv.management.base.AMX;
035    import com.sun.appserv.management.base.Container;
036    
037    /**
038            The discovery and navigation of all managed objects in the J2EE
039            management system begins with the J2EEDomain.
040    
041            @see J2EEServer
042            @see J2EECluster
043            @see J2EEApplication
044            @see JVM
045            @see AppClientModule
046            @see EJBModule
047            @see WebModule
048            @see ResourceAdapterModule
049            @see EntityBean
050            @see StatefulSessionBean
051            @see StatelessSessionBean
052            @see MessageDrivenBean
053            @see Servlet
054            @see JavaMailResource
055            @see JCAResource
056            @see JCAConnectionFactory
057            @see JCAManagedConnectionFactory
058            @see JDBCResource
059            @see JDBCDataSource
060            @see JDBCDriver
061            @see JMSResource
062            @see JNDIResource
063            @see JTAResource
064            @see RMIIIOPResource
065            @see URLResource
066     */
067    public interface J2EEDomain
068            extends J2EEManagedObject,ConfigProvider, Container
069    {
070            public final static String              J2EE_TYPE       = J2EETypes.J2EE_DOMAIN;
071            
072            /**
073                    Note that the Attribute name is case-sensitive
074                    "servers" as defined by JSR 77.
075                    
076                    @return the ObjectNames of the J2EEServers, as Strings
077             */
078            public String[] getservers();
079    
080    
081            /**
082                    @return a Map keyed by server name of J2EEServer
083                    @deprecated used {@link #getJ2EEServerMap}
084             */
085            public Map<String,J2EEServer>             getServerMap();
086            
087    
088            /**
089                    @return a Map keyed by server name of J2EEServer
090             */
091            public Map<String,J2EEServer>             getJ2EEServerMap();
092            
093            /**
094                    @return the ObjectNames of the J2EEClusters, as Strings
095                    @deprecated use {@link #getJ2EEClusterMap}
096             */
097            public String[] getclusters();
098            
099             /**
100           @return a Map keyed by cluster name of J2EECluster
101               @deprecated use {@link #getJ2EEClusterMap}
102        */
103       public Map<String,J2EECluster>        getClusterMap(); 
104       
105             /**
106           @return a Map keyed by cluster name of J2EECluster
107        */
108       public Map<String,J2EECluster>        getJ2EEClusterMap(); 
109    }
