030    package com.sun.appserv.management.monitor.statistics;
031    import javax.management.j2ee.statistics.Stats;
032    import javax.management.j2ee.statistics.CountStatistic;
033    import javax.management.j2ee.statistics.BoundedRangeStatistic;
034    
035    /** A Stats interface to represent the statistical data exposed by an EJB Bean Pool.
036     * These are based on the statistics exposed in S1AS7.0. 
037     * All the EJB Pool implementations should expose statistical data by implementing this interface.
038     */
039    
040    public interface EJBPoolStats extends Stats
041    {
042    
043            /** Returns the statistical information about the number of EJBs in the associated pool, as an instance of BoundedRangeStatistic.
044             * This returned value gives an idea about how the pool is changing.
045             * @return              an instance of {@link BoundedRangeStatistic}
046         */
047        public BoundedRangeStatistic getNumBeansInPool();
048        
049        /** Returns the number of threads waiting for free Beans, as an instance of CountStatistic.
050             * This indicates the congestion of requests.
051             * @return              an instance of {@link BoundedRangeStatistic}
052         */
053        public BoundedRangeStatistic getNumThreadsWaiting();
054        
055        /** Returns the number of Beans created in associated pool so far over time, since the gathering of data started, as a CountStatistic.
056             * @return              an instance of {@link CountStatistic}
057         */
058        public CountStatistic getTotalBeansCreated();
059        
060        /** Returns the number of Beans destroyed from associated pool so far over time, since the gathering of data started, as a CountStatistic.
061             * @return              an instance of {@link CountStatistic}
062         */
063        public CountStatistic getTotalBeansDestroyed();
064        
065        /** Returns the maximum number of messages to load into a JMS session, at a time, as a CountStatistic.
066             * @return              an instance of {@link CountStatistic}
067         */
068        public CountStatistic getJMSMaxMessagesLoad();
069    }
