005    package amxsamples;
006    
007    import java.util.Map;
008    
009    import javax.management.NotificationListener;
010    import javax.management.Notification;
011    
012    import com.sun.appserv.management.deploy.DeploymentStatus;
013    import com.sun.appserv.management.deploy.DeploymentMgr;
014    import com.sun.appserv.management.deploy.DeploymentSupport;
015    import com.sun.appserv.management.deploy.DeploymentProgress;
016    
017    
018    /**
019            A JMX NotificationListener for deployment.
020     */
021    public final class DeployNotificationListener
022                    implements NotificationListener
023    {
024            private final Object            mDeployID;
025            private boolean                         mIsCompleted;
026            private DeploymentStatus        mDeploymentStatus;
027            
028                    public
029            DeployNotificationListener( final Object        deployID )
030            {
031                    mDeployID               = deployID;
032                    mIsCompleted    = false;
033            }
034            
035                    public boolean
036            isCompleted()
037            {
038                    return( mIsCompleted );
039            }
040            
041                    public DeploymentStatus
042            getDeploymentStatus()
043            {
044                    return( mDeploymentStatus );
045            }
046            
047                    public synchronized void
048            handleNotification(
049                    final Notification      notif, 
050                    final Object            handback) 
051            {
052                    try
053                    {
054                            realHandleNotification( notif, handback );
055                    }
056                    catch( Exception e )
057                    {
058                            e.printStackTrace();
059                    }
060            }
061            
062            /**
063                    Note that Notifications are not guaranteed to be delivered in order.
064                    Thus, it is theoretically possible for a DEPLOYMENT_COMPLETED_NOTIFICATION_TYPE
065                    to be received before a DEPLOYMENT_STARTED_NOTIFICATION_TYPE.
066             */
067                    public void
068            realHandleNotification(
069                    final Notification      notif, 
070                    final Object            handback) 
071            {
072                    final String    type    = notif.getType();
073                    final Map               m       = (Map)notif.getUserData();
074                    final Object    deployID        = m.get( DeploymentMgr.NOTIF_DEPLOYMENT_ID_KEY );
075                    
076                    if ( deployID.equals( mDeployID ) )
077                    {
078                            if ( type.equals( DeploymentMgr.DEPLOYMENT_STARTED_NOTIFICATION_TYPE ) )
079                            {
080                                    SampleUtil.println( "Deployment started for " + deployID);
081                            }
082                            else if ( type.equals( DeploymentMgr.DEPLOYMENT_COMPLETED_NOTIFICATION_TYPE ) )
083                            {
084                                    final Map       statusData      = (Map)
085                                            m.get( DeploymentMgr.NOTIF_DEPLOYMENT_COMPLETED_STATUS_KEY );
086                                    
087                                    final DeploymentStatus  status  = 
088                                            DeploymentSupport.mapToDeploymentStatus( statusData );
089                                    
090                                    SampleUtil.println( "Deployment completed for " + deployID + " with status: " + 
091                                            status.getStageStatus() );
092                                    
093                                    mIsCompleted    = true;
094                                    mDeploymentStatus       = status;
095                            }
096                            else if ( type.equals( DeploymentMgr.DEPLOYMENT_PROGRESS_NOTIFICATION_TYPE ) )
097                            {
098                                    final Map       progressData    = (Map)
099                                            m.get( DeploymentMgr.NOTIF_DEPLOYMENT_PROGRESS_KEY );
100                                    
101                                    final DeploymentProgress        progress        = 
102                                            DeploymentSupport.mapToDeploymentProgress( progressData );
103                                            
104                                    SampleUtil.println( "Deployment progress for " + deployID + " = " + 
105                                            progress.getProgressPercent() + "%" );
106                            }
107                    }
108            }
109    }
