023    package com.sun.appserv.management.config;
024    
025    import java.util.Map;
026    import com.sun.appserv.management.base.Container;
027    import com.sun.appserv.management.base.AMX;
028    import com.sun.appserv.management.base.XTypes;
029    
030    /**
031     * Configuration for the &lt;group-management-service&gt; for a &lt;config&gt; in
032     * a domain. This element controls the behavior of the group management service
033     * used for cluster monitoring and failure detection.
034     * @since Appserver 9.0
035    */
036    
037    public interface GroupManagementServiceConfig
038        extends AMX, ConfigElement, PropertiesAccess
039    {
040        /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
041        public static final String  J2EE_TYPE    = XTypes.GROUP_MANAGEMENT_SERVICE_CONFIG;
042    
043        /** Return the FD protocol tries.
044         *  This is the maximum number of attempts to try before GMS confirms that a  
045         *  failure is suspected in the group.
046         *  @return positive integer specifying the number of such attempts
047         */
048        public String getFDProtocolMaxTries();
049        /** Set the FD protocol tries to the specified positive integer value.
050          * Must be a positive integer.
051          * @param tries a positive integer specifying the number of attempts
052        */
053        public void setFDProtocolMaxTries(final String tries);
054    
055    
056        public String getFDProtocolTimeoutMillis();
057        public void setFDProtocolTimeoutMillis(final String duration);
058    
059        public String getMergeProtocolMaxIntervalMillis();
060        public void setMergeProtocolMaxIntervalMillis(final String duration);
061    
062        public String getMergeProtocolMinIntervalMillis();
063        public void setMergeProtocolMinIntervalMillis(final String duration);
064    
065        public String getPingProtocolTimeoutMillis();
066        public void setPingProtocolTimeoutMillis(final String duration);
067    
068        public String getVSProtocolTimeoutMillis();
069        public void setVSProtocolTimeoutMillis(final String duration);
070    }
