025    package com.sun.appserv.management.config;
026    
027    import java.util.Map;
028    
029    /**
030            Supports creation and removal of server references.
031    */
032    public interface ClusterRefConfigCR extends ConfigRemover
033    {
034        public final static String  LB_POLICY_KEY           = "LBPolicy";
035    
036        public final static String  LB_POLICY_MODULE_KEY    = "LBPolicyModule";
037    
038        /**
039          Creates a new cluster-ref for an lb-config.
040          @param clusterName A reference to the name of a cluster.
041          @param reserved   reserved for future use
042          @return A proxy to the ClusterRefConfig MBean. 
043         */
044        public ClusterRefConfig createClusterRefConfig(
045            String clusterName, Map<String,String> reserved);
046    
047        /**
048        Creates a new cluster-ref.
049    
050        * @param referencedClusterName name of the cluster.
051        * @param lbPolicy load balancing policy for this cluster
052        * @param lbPolicyModule absolute path to the shared library implementing 
053        * the user-defined policy.
054        * @return A proxy to the ClusterRefConfig MBean.
055        */
056        public ClusterRefConfig createClusterRefConfig(
057            String referencedClusterName, String lbPolicy, String lbPolicyModule);
058        /**
059          Removes a cluster-ref for an lb-config.
060          @param clusterName A reference to the name of a cluster.
061         */
062        public void removeClusterRefConfig(String clusterName );
063    }
