023    package com.sun.appserv.management.j2ee;
024    
025    import java.util.Map;
026    import java.util.Set;
027    
028    import javax.management.ObjectName;
029    
030    import com.sun.appserv.management.base.Container;
031    
032    
033    /**
034     */
035    public interface J2EEServer extends J2EELogicalServer, Container
036    {
037            public final static String              J2EE_TYPE       = J2EETypes.J2EE_SERVER;
038            
039            /**
040                Restart the server.
041                <b>Enterprise Edition only.</b>
042             */
043            public void restart();
044            
045            /**
046                    Note that the Attribute name is case-sensitive
047                    "deployedObjects" as defined by JSR 77.
048                    
049                    @return the ObjectNames as Strings
050             */
051            public String[] getdeployedObjects();
052            
053            /**
054                    Return Set of all deployed objects.  Possible types include:
055                    <ul>
056                    <li>{@link com.sun.appserv.management.j2ee.J2EEApplication}</li>
057                    <li>{@link com.sun.appserv.management.j2ee.WebModule}</li>
058                    <li>{@link com.sun.appserv.management.j2ee.EJBModule}</li>
059                    <li>{@link com.sun.appserv.management.j2ee.AppClientModule}</li>
060                    <li>{@link com.sun.appserv.management.j2ee.ResourceAdapterModule}</li>
061                    </ul>
062                    <p>
063                    To obtain Resources of a particular type, use
064                    {@link Container#getContaineeMap}(j2eeType).
065             */
066            public <T extends J2EEDeployedObject> Set<T>        getDeployedObjectsSet();
067            
068            /**
069                    In 8.1, there will only ever be one JVM for a J2EEServer.
070                    Note that the Attribute name is case-sensitive
071                    "javaVMs" as defined by JSR 77.
072                    
073                    @return the ObjectNames as Strings
074             */
075            public String[] getjavaVMs();
076            
077            /**
078                    There is always a single JVM for a J2EEServer.
079                    @return JVM
080             */
081            public JVM              getJVM();
082            
083            /**
084                    Note that the Attribute name is case-sensitive
085                    "resources" as defined by JSR 77.
086                    
087                    @return the ObjectNames as Strings
088             */
089            public String[]         getresources();
090            
091            /**
092                    Return Set of all resources.  Possible types include:
093                    <ul>
094                    <li>{@link com.sun.appserv.management.j2ee.JDBCResource}</li>
095                    <li>{@link com.sun.appserv.management.j2ee.JavaMailResource}</li>
096                    <li>{@link com.sun.appserv.management.j2ee.JCAResource}</li>
097                    <li>{@link com.sun.appserv.management.j2ee.JMSResource}</li>
098                    <li>{@link com.sun.appserv.management.j2ee.JNDIResource}</li>
099                    <li>{@link com.sun.appserv.management.j2ee.JTAResource}</li>
100                    <li>{@link com.sun.appserv.management.j2ee.RMIIIOPResource}</li>
101                    <li>{@link com.sun.appserv.management.j2ee.URLResource}</li>
102                    </ul>
103                    <p>
104                    To obtain Resources of a particular type, use
105                    {@link Container#getContaineeMap}(j2eeType).
106             */
107            public <T extends J2EEResource> Set<T> getResourcesSet();
108            
109            /**
110                    Note that the Attribute name is case-sensitive
111                    "serverVendor" as defined by JSR 77.
112                    
113                    @return the server vendor, a free-form String
114             */
115            public String           getserverVendor();
116            
117            /**
118                    Note that the Attribute name is case-sensitive
119                    "serverVersion" as defined by JSR 77.
120                    
121                    @return the server version, a free-form String
122             */
123            public String           getserverVersion();
124            
125    
126            /**
127                    @return true if server configuration has changed such that a restart must
128                    be performed.
129             */
130        public boolean  getRestartRequired();
131    }
