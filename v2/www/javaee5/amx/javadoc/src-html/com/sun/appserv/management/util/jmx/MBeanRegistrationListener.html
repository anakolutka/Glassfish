023    package com.sun.appserv.management.util.jmx;
024    
025    import java.io.IOException;
026    
027    import javax.management.ObjectName;
028    import javax.management.Notification;
029    import javax.management.NotificationListener;
030    import javax.management.NotificationFilter;
031    import javax.management.MBeanServerNotification;
032    import javax.management.MBeanServerConnection;
033    import javax.management.InstanceNotFoundException;
034    import com.sun.appserv.management.base.AMXRootLogger;
035    
036    /**
037            Convenience base class for listening to 
038            {@link MBeanServerNotification} notifications.
039            A class extending this class must implement {@link #mbeanRegistered}
040            and {@link #mbeanUnregistered}.
041            <p>
042            The class is designed to start listening upon creation.
043            The caller should call cleanup() when listening is no longer
044            desired.  Once cleanup() is called, no further listening can
045            be done; a new MBeanRegistrationListener should be instantiated
046            if further listening is desired.
047     */
048    public abstract class MBeanRegistrationListener extends NotificationListenerBase
049    {
050        private final ObjectName    mRegUnregFilter;
051        private final String        mDefaultDomain;
052        
053            /**
054                If 'constrain' is non-null, then all registration and unregistration
055                events will be filtered through it.  Only those MBeans
056                matching will be passed through to {@link #mbeanRegistered}
057                and {@link #mbeanUnregistered}.
058                
059                @param conn
060                @param constrain     optional fixed or pattern ObjectName
061             */
062                public
063            MBeanRegistrationListener(
064                final MBeanServerConnection conn,
065                final ObjectName    constrain )
066                    throws InstanceNotFoundException, IOException
067            {
068                super( conn,
069                    JMXUtil.getMBeanServerDelegateObjectName() );
070                mRegUnregFilter = constrain;
071                
072                mDefaultDomain  = conn.getDefaultDomain();
073            }
074            
075            /**
076                Calls this( conn, null ).
077                @param conn
078             */
079               public
080            MBeanRegistrationListener( final MBeanServerConnection conn)
081                    throws InstanceNotFoundException, IOException
082            {
083                this( conn, (ObjectName)null );
084            }
085        
086        protected abstract void mbeanRegistered( final ObjectName objectName );
087        protected abstract void mbeanUnregistered( final ObjectName objectName );
088            
089            public void
090            handleNotification( final Notification notifIn, final Object handback)
091            {
092                    if ( ! (notifIn instanceof MBeanServerNotification) )
093                    {
094                        throw new IllegalArgumentException( notifIn.toString() );
095                    }
096                    
097                    final MBeanServerNotification notif     = (MBeanServerNotification)notifIn;
098                    final ObjectName objectName     = notif.getMBeanName();
099                    final String     type   = notif.getType();
100                    
101                    final boolean matchesFilter   = (mRegUnregFilter == null) ||
102                            JMXUtil.matchesPattern( mDefaultDomain, mRegUnregFilter, objectName );
103                            
104                    if ( matchesFilter )
105                    {
106                if ( type.equals( MBeanServerNotification.REGISTRATION_NOTIFICATION  ) )
107                {
108                    mbeanRegistered( objectName );
109                }
110                else if ( type.equals( MBeanServerNotification.UNREGISTRATION_NOTIFICATION  ) )
111                {
112                    mbeanUnregistered( objectName );
113                }
114            }
115            }
116    }
