023    package com.sun.appserv.management.util.jmx.stringifier;
024    
025    import java.lang.reflect.Array;
026    import javax.management.MBeanInfo;
027    import javax.management.MBeanOperationInfo;
028    import javax.management.MBeanAttributeInfo;
029    import javax.management.MBeanConstructorInfo;
030    import javax.management.MBeanNotificationInfo;
031    
032    import com.sun.appserv.management.util.stringifier.Stringifier;
033    import com.sun.appserv.management.util.stringifier.ArrayStringifier;
034    
035    public class MBeanInfoStringifier extends MBeanFeatureInfoStringifier implements Stringifier
036    {
037            public static final MBeanInfoStringifier        DEFAULT = new MBeanInfoStringifier();
038            
039                    public
040            MBeanInfoStringifier()
041            {
042                    super();
043            }
044            
045                    public
046            MBeanInfoStringifier( MBeanFeatureInfoStringifierOptions options )
047            {
048                    super( options );
049            }
050            
051                    private String
052            stringifyArray( Object [] a, Stringifier stringifier)
053            {
054                    String  temp    = "";
055                    
056                    if ( a.length != 0 )
057                    {
058                            temp    = "\n" + ArrayStringifier.stringify( a, "\n", stringifier);
059                    }
060                    return( temp );
061            }
062            
063            // subclass may override
064                    MBeanAttributeInfoStringifier
065            getMBeanAttributeInfoStringifier( MBeanFeatureInfoStringifierOptions options )
066            {
067                    return( new MBeanAttributeInfoStringifier(options) );
068            }
069            
070            // subclass may override
071                    MBeanOperationInfoStringifier
072            getMBeanOperationInfoStringifier( MBeanFeatureInfoStringifierOptions options )
073            {
074                    return( new MBeanOperationInfoStringifier(options) );
075            }
076            
077            // subclass may override
078                    MBeanConstructorInfoStringifier
079            getMBeanConstructorInfoStringifier( MBeanFeatureInfoStringifierOptions options )
080            {
081                    return( new MBeanConstructorInfoStringifier(options) );
082            }
083            
084            // subclass may override
085                    MBeanNotificationInfoStringifier
086            getMBeanNotificationInfoStringifier( MBeanFeatureInfoStringifierOptions options )
087            {
088                    return( new MBeanNotificationInfoStringifier(options) );
089            }
090            
091                    public String
092            stringify( Object o)
093            {
094                    String                  result  = "";
095                    final MBeanInfo info    = (MBeanInfo)o;
096                    
097                    final MBeanOperationInfo []             operations              = info.getOperations();
098                    final MBeanAttributeInfo []             attributes              = info.getAttributes();
099                    final MBeanConstructorInfo []   constructors    = info.getConstructors();
100                    final MBeanNotificationInfo []  notifications   = info.getNotifications();
101                    final String    description = info.getDescription();
102                    
103                    result  = "Summary: " +
104                                            operations.length + " operations, " +
105                                            attributes.length + " attributes, " +
106                                            constructors.length + " constructors, " +
107                                            notifications.length + " notifications" +
108                                            (description == null ? "" : ", \"" + description + "\"" );
109                    
110                    final MBeanFeatureInfoStringifierOptions options        =
111                                    new MBeanFeatureInfoStringifierOptions( true, "," );
112                    
113                    // Do formal terms like "Attributes" need to be I18n?
114                    // Probabably not as they are part of a specification.
115                    result  = result + "\n\n- Attributes -" +
116                                            stringifyArray( attributes, getMBeanAttributeInfoStringifier( options ) );
117            
118                    result  = result + "\n\n- Operations -" +
119                                            stringifyArray( operations, getMBeanOperationInfoStringifier(options) );
120            
121                    result  = result + "\n\n- Constructors -" +
122                                            stringifyArray( constructors, getMBeanConstructorInfoStringifier(options) );
123            
124                    result  = result + "\n\n- Notifications -" + 
125                                            stringifyArray( notifications, getMBeanNotificationInfoStringifier(options) );
126                    
127                    return( result );
128                            
129            }
130    }
