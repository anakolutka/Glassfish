030    package com.sun.appserv.management.monitor.statistics;
031    import javax.management.j2ee.statistics.Stats;
032    import javax.management.j2ee.statistics.CountStatistic;
033    import javax.management.j2ee.statistics.TimeStatistic;
034    
035    /** A Stats interface to represent the statistical data exposed by an EJB Business Method.
036     * These are based on the statistics exposed in S1AS7.0. 
037     * All the EJB Methods should expose statistical data by implementing this interface.
038     */
039    
040    public interface EJBMethodStats extends Stats
041    {
042        
043            /** Returns the statistics of the method invocation as an instance of TimeStatistic.
044             * Note that it returns the number of times the operation called, the total time
045             * that was spent during the invocation and so on. All the calculations of the
046             * statistic are being done over time.
047             * @return in instance of {@link TimeStatistic}
048             */
049            public TimeStatistic getMethodStatistic();
050            
051        /** Returns the total number of errors as a CountStatistic. It is upto the method
052             * implementor to characterize what an error is. Generally if an operation results in 
053             * an exception, this count will increment by one.
054             * @return              an instance of {@link CountStatistic}
055         */
056        public CountStatistic getTotalNumErrors();
057        
058        /** Returns the total number of successful runs, as a CountStatistic. It is upto the method
059             * implementor to characterize what a successful run is. Generally if an operation returns
060             * normally, this count will increment by one.
061             * @return              an instance of {@link CountStatistic}
062         */
063        public CountStatistic getTotalNumSuccess();
064        
065        /** Returns the time spent during the last successful/unsuccessful attempt to execute the operation, as a CountStatistic.
066             * The time spent is generally an indication of the system load/processing time.
067             * @return              an instance of {@link CountStatistic}
068         */
069        public CountStatistic getExecutionTime();    
070    }
