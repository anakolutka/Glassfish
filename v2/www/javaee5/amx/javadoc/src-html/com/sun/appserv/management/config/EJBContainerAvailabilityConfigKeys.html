032    /**
033            Parameters used by {@link AvailabilityServiceConfig}

035            
036            @see AvailabilityServiceConfig
037            @see SFSBPersistenceTypeValues
038     */

040    {
041            private EJBContainerAvailabilityConfigKeys()    {}
042            
043            public static final String EJB_CONTAINER_AVAILABILITY_ENABLED_KEY       = "AvailabilityEnabled";

045            /**
046                    Legal values are those defined in {@link SFSBPersistenceTypeValues}.
047             */
048            public static final String SFSB_HA_PERSISTENCE_TYPE_KEY = "SFSBHAPersistenceType";
049            
050            /**
051                    See {@link #SFSB_HA_PERSISTENCE_TYPE_KEY}.
052             */
053            public static final String SFSB_PERSISTENCE_TYPE_KEY            = "SFSBPersistenceType";
054    
055            public static final String SFSB_CHECKPOINT_ENABLED_KEY          = "SFSBCheckPointEnabled";
056    
057            public static final String SFSB_QUICK_CHECKPOINT_ENABLED_KEY            = "SFSBQuickCheckPointEnabled";
058    
059            /**
060                    The jndi-name for the JDBC Connection Pool used       
061            potentially by both the Web Container and the EJB Stateful    
062            Session Bean Container for use in checkpointing/passivation   
063            when persistence-type = {@link SFSBPersistenceTypeValues#HA}.
064             */
065            public static final String SFSB_STORE_POOL_NAME_KEY             = "SFSBStorePoolName";
066    }
