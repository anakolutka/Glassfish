024    package com.sun.appserv.management.util.jmx;
025    
026    import javax.management.MBeanOperationInfo;
027    
028    import com.sun.appserv.management.util.jmx.stringifier.MBeanOperationInfoStringifier;
029    import com.sun.appserv.management.util.jmx.stringifier.MBeanFeatureInfoStringifierOptions;
030    
031    
032    /**
033            Caution: this Comparator may be inconsistent with equals() because it ignores the description.
034     */
035    public final class MBeanOperationInfoComparator
036        implements java.util.Comparator<MBeanOperationInfo>
037    {
038            private static final MBeanOperationInfoStringifier              OPERATION_INFO_STRINGIFIER      =
039                    new MBeanOperationInfoStringifier( new MBeanFeatureInfoStringifierOptions( false, ",") );
040                    
041                    
042            public static final MBeanOperationInfoComparator
043                INSTANCE    = new MBeanOperationInfoComparator();
044            
045            private MBeanOperationInfoComparator()  {}
046            
047                    public int
048            compare( final MBeanOperationInfo info1, final MBeanOperationInfo info2 )
049            {
050                    final MBeanOperationInfoStringifier     sf      = OPERATION_INFO_STRINGIFIER;
051                    
052                    // we just want to sort based on name and signature; there can't be two operations with the
053                    // same name and same signature, so as long as we include the name and signature the
054                    // sorting will always be consistent.
055                    int     c       = info1.getName().compareTo( info2.getName() );
056                    if ( c == 0 )
057                    {
058                            // names the same, subsort on signature, first by number of params
059                            c       = info1.getSignature().length - info2.getSignature().length;
060                            if ( c == 0 )
061                            {
062                                    // names the same, subsort on signature, first by number of params
063                                    c       = sf.getSignature( info1 ).compareTo( sf.getSignature( info2 ) );
064                            }
065                            
066                    }
067                    
068                    return( c );
069            }
070            
071                    public boolean
072            equals( Object other )
073            {
074                    return( other instanceof MBeanOperationInfoComparator );
075            }
076    }
