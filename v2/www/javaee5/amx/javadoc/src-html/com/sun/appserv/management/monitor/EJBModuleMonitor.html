027    package com.sun.appserv.management.monitor;
028    
029    import java.util.Map;
030    
031    import com.sun.appserv.management.base.XTypes;
032    import com.sun.appserv.management.base.Container;
033    
034    
035    public interface EJBModuleMonitor extends Container, Monitoring
036    {
037    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
038            public static final String      J2EE_TYPE       = XTypes.EJB_MODULE_MONITOR;
039            
040            /**
041                    @return Map of {@link StatelessSessionBeanMonitor} keyed by name (beanName)
042             */
043            public Map<String,StatelessSessionBeanMonitor>    getStatelessSessionBeanMonitorMap();
044            
045            /**
046                    @return Map of {@link StatefulSessionBeanMonitor} keyed by name (beanName)
047             */
048            public Map<String,StatefulSessionBeanMonitor>     getStatefulSessionBeanMonitorMap();
049            
050            /**
051                    @return Map of {@link EntityBeanMonitor} keyed by name (beanName)
052             */
053            public Map<String,EntityBeanMonitor>      getEntityBeanMonitorMap();
054            
055            /**
056                    @return Map of {@link MessageDrivenBeanMonitor} keyed by name (beanName)
057             */
058            public Map<String,MessageDrivenBeanMonitor>       getMessageDrivenBeanMonitorMap();
059            
060            /**
061                    @return Map of {@link WebServiceEndpointMonitor} keyed by name 
062             */
063            public Map<String,WebServiceEndpointMonitor>      getWebServiceEndpointMonitorMap();
064    }
