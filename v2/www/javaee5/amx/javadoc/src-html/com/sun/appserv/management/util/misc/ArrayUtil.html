023    package com.sun.appserv.management.util.misc;
024    
025    import java.lang.reflect.Array;
026    
027    /**
028            Provides:
029            - utility to check for equality
030     */
031    public final class ArrayUtil
032    {
033                    private
034            ArrayUtil( )
035            {
036                    // disallow instantiation
037            }
038            
039                    public static boolean
040            arraysEqual( final Object array1, final Object array2 )
041            {
042                    boolean equal   = array1 == array2;
043                    
044                    if ( equal )
045                    {
046                            // same object or both null
047                            return( true );
048                    }
049                    else if ( array1 == null || array2 == null)
050                    {
051                            return( false );
052                    }
053                    
054    
055                    if ( array1.getClass() == array2.getClass() &&
056                            ClassUtil.objectIsArray( array1 ) &&
057                            Array.getLength( array1 ) == Array.getLength( array2 ) )
058                    {
059                            equal   = true;
060                            final int       length  = Array.getLength( array1 );
061                            
062                            for( int i = 0; i < length; ++ i )
063                            {
064                                    final Object    a1      = Array.get( array1, i );
065                                    final Object    a2      = Array.get( array2, i );
066                                    
067                                    if ( a1 != a2)
068                                    {
069                                            if ( a1 == null || a2 == null )
070                                            {
071                                                    equal   = false;
072                                            }
073                                            else if ( ClassUtil.objectIsArray( a1 )  )
074                                            {
075                                                    if ( ! arraysEqual( a1, a2 ) )
076                                                    {
077                                                            equal   = false;
078                                                    }
079                                            }
080                                    }
081    
082                                    if ( ! equal )
083                                            break;
084                            }
085                    }
086                    
087                    return( equal );
088            }
089    
090                    public static boolean
091            arrayContainsNulls( final Object[] array )
092            {
093                    boolean containsNulls   = false;
094                    
095                    for( int i = 0; i < array.length; ++i )
096                    {
097                            if ( array[ i ] == null )
098                            {
099                                    containsNulls   = true;
100                                    break;
101                            }
102                    }
103                    
104                    return( containsNulls );
105            }
106            
107    
108            public static <T> T[]
109        newArray( final Class<T> theClass, int numItems )
110        {
111            return (T[])( Array.newInstance( theClass, numItems ) );
112        }
113            
114            /**
115                    Create a new array from the original.
116                    
117                    @param items            the original array
118                    @param startIndex       index of the first item
119                    @param numItems         
120                    @return an array of the same type, containing numItems items
121             */
122                    public static <T> T[]
123            newArray(
124                    final T[]       items,
125                    final int       startIndex,
126                    final int       numItems )
127            {
128                    final Class<T>    theClass        = TypeCast.asClass( items.getClass().getComponentType() );
129                    
130                    final T[]       result  = newArray( theClass, numItems );
131                    System.arraycopy( items, startIndex, result, 0, numItems );
132                    
133                    return( result );
134            }
135            
136            /**
137                    Create a new array consisting of originals and new.
138                    
139                    @param items1           1st array
140                    @param items2           2nd array
141                    @return an array of the same type as items1, its elements first
142             */
143                    public static <T> T[]
144            newArray(
145                    final T[]       items1,
146                    final T[]       items2)
147            {
148                    final Class<T> class1     = TypeCast.asClass( items1.getClass().getComponentType() );
149                    final Class<T> class2     = TypeCast.asClass( items2.getClass().getComponentType() );
150                    
151                    if ( ! class1.isAssignableFrom( class2 ) )
152                    {
153                            throw new IllegalArgumentException();
154                    }
155                    
156                    final int       length1 = Array.getLength( items1 );
157                    final int       length2 = Array.getLength( items2 );
158                    final T[]       result  = newArray( class1, length1 + length2 );
159                    System.arraycopy( items1, 0, result, 0, length1 );
160                    System.arraycopy( items2, 0, result, length1, length2 );
161                    
162                    return( result );
163            }
164            
165            /**
166                    Create a new array consisting of an original array, and a single new item.
167                    
168                    @param items            an array
169                    @param item             an item to append
170                    @return an array of the same type as items1, its elements first
171             */
172                    public static <T> T[]
173            newArray(
174                    final T[]       items,
175                    final T item)
176            {
177                    final Class<T>    theClass        = TypeCast.asClass( items.getClass().getComponentType() );
178                    final T[]       result  = newArray( theClass, items.length + 1 );
179                    System.arraycopy( items, 0, result, 0, items.length );
180                    
181                    result[ result.length - 1 ]     = item;
182                    return( result );
183            }
184    }
