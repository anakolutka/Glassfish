030    package com.sun.appserv.management.j2ee.statistics;
031    
032    import java.util.Map;
033    import java.io.Serializable;
034    
035    import javax.management.openmbean.CompositeData;
036    
037    import javax.management.j2ee.statistics.Statistic;
038    import javax.management.j2ee.statistics.CountStatistic;
039    
040    import com.sun.appserv.management.util.jmx.OpenMBeanUtil;
041    import com.sun.appserv.management.util.misc.ObjectUtil;
042    
043    
044    /**
045            
046     */
047    public class CountStatisticImpl extends StatisticImpl
048            implements CountStatistic, Serializable
049    {
050            static final long serialVersionUID = -4868791714488583778L;
051            
052            /* member name as defined by JSR 77 */
053            private final long      Count;
054            
055                    public
056            CountStatisticImpl(
057                    final String    name,
058                    final String    description,
059                    final String    unit,
060                    final long              startTime,
061                    final long              lastSampleTime,
062                    final long              count )
063            {
064                    super( name, description, unit, startTime, lastSampleTime );
065                    Count   = count;
066            }
067            
068                    public
069            CountStatisticImpl( final CompositeData compositeData )
070            {
071                    this( OpenMBeanUtil.compositeDataToMap( compositeData ) );
072            }
073            
074                    public
075            CountStatisticImpl( final CountStatistic s )
076            {
077                    super( s );
078                    Count   = s.getCount();
079            }
080            
081                    public
082            CountStatisticImpl( final MapStatistic s )
083            {
084                    super( s );
085                    Count   = s.getlong( "Count" );
086            }
087            
088                    public
089            CountStatisticImpl( final Map<String,?> data )
090            {
091                    this( new MapStatisticImpl( data ) );
092            }
093    
094    
095                    public long
096            getCount()
097            {
098                    return( Count );
099            }
100            
101                public int
102            hashCode()
103            {
104                return super.hashCode() ^ ObjectUtil.hashCode( Count );
105            }
106            
107            
108                    public boolean
109            equals( final Object rhs )
110            {
111                    boolean equals  = super.equals( rhs ) && (rhs instanceof CountStatistic);
112                    
113                    if ( equals )
114                    {
115                            final CountStatistic    s       = (CountStatistic)rhs;
116                            
117                            equals  = getCount() == s.getCount();
118                    }
119                    return( equals );
120            }
121    }
