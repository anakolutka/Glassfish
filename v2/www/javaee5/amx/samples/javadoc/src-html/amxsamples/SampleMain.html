051                    public static void
052            main( final String[] args )
053            {
054                    if ( args.length > 1 )
055                    {
056                            SampleUtil.println( "Specify a properties file or nothing." );
057                            System.exit( 255 );
058                    }
059                    new StringifierRegistryIniter( StringifierRegistryImpl.DEFAULT );
060                    
061                    checkForMissingClasses();
062                    
063                    try
064                    {
065                            new SampleMain( args.length == 1 ? args[ 0 ] : "SampleMain.properties" );
066                    }
067                    catch( final Throwable t )
068                    {
069                        final Throwable rootCause   = ExceptionUtil.getRootCause( t );
070                        
071                        if ( rootCause instanceof ConnectException )
072                        {
073                            SampleUtil.println( "Unable to connect to specified host and port." );
074                        }
075                        else
076                        {
077                                rootCause.printStackTrace();
078                            }
079                    }
080            }

152                    private void
153            handleChoice(
154                    final Samples   samples,
155                    final String    line )
156                    throws IOException, InstanceNotFoundException
157            {
158                    final String[]  parts   = line.split( "[ \t]+" );
159                    final int               numArgs = parts.length - 1;
160                    final String    cmd     = parts[ 0 ];
161                    
162                    final String[] args    = ArrayUtil.newArray( parts, 1, numArgs );
163                    
164                    if ( cmd.length() != 0 )
165                    {
166                            SampleUtil.println( "cmd: " + SampleUtil.arrayToString( parts, " ", "" ) );
167                    }
168                    
169                    if ( cmd.equals( QUIT ) || cmd.equals( "q" ) )
170                    {
171                            require( numArgs == 0, "Usage: " + QUIT );
172                            System.exit( 0 );
173                    }
174                    if ( cmd.equalsIgnoreCase( HELP ) || cmd.equals( "?" ) )
175                    {
176                            SampleUtil.println( COMMANDS );
177                    }
178                    else if ( cmd.length() == 0 )
179                    {
180                            // do nothing
181                    }
182                    else if ( cmd.equals( DEPLOY ) )
183                    {
184                            require( numArgs >= 1, "Usage: " + DEPLOY + " <archive-name>" );
185                            for( int i = 1; i < parts.length; ++i )
186                            {
187                                    samples.deploy( new File( parts[ i ] ) );
188                            }
189                    }
190                    else if ( cmd.equals( UNDEPLOY ) )
191                    {
192                            require( numArgs >= 1, "Usage: " + UNDEPLOY + " [<name>[ <name>]*]" );
193                            for( int i = 1; i < parts.length; ++i )
194                            {
195                                    samples.undeploy( parts[ i ] );
196                            }
197                    }
198                    else if ( cmd.equals( START_SERVER ) )
199                    {
200                            require( numArgs == 1, "Usage: " + START_SERVER + " <server-name>" );
201                            samples.startServer( parts[ 1 ] );
202                    }
203                    else if ( cmd.equals( STOP_SERVER ) )
204                    {
205                            require( numArgs == 1, "Usage: " + STOP_SERVER + " <server-name>" );
206                            samples.stopServer( parts[ 1 ] );
207                    }
208                    else if ( cmd.equals( LIST ) )
209                    {
210                            require( numArgs == 0, "Usage: " + LIST );
211                            samples.handleList();
212                    }
213                    else if ( cmd.equals( SHOW_HIERARCHY ) )
214                    {
215                            if ( numArgs == 0 )
216                            {
217                                    samples.displayHierarchy( false );
218                            }
219                            else
220                            {
221                                int count   = parts.length;
222                                final String last   = parts[ count - 1];
223                                
224                                boolean showAttrs   = false;
225                                if ( last.equalsIgnoreCase( "true" ) ||
226                                        last.equalsIgnoreCase( "false" ) )
227                                {
228                                    SampleUtil.println( "XXX" );
229                                    showAttrs   = Boolean.valueOf( last );
230                                    --count;
231                                }
232                                
233                                if ( count >= 2 )
234                                {
235                                    for( int i = 1; i < count; ++i )
236                                    {
237                                            samples.displayHierarchy( parts[ i ], showAttrs );
238                                    }
239                                    }
240                                    else
241                                    {
242                                        samples.displayHierarchy( showAttrs );
243                                    }
244                            }
245                    }
246                    else if ( cmd.equals( QUERY) )
247                    {
248                            require( numArgs == 0, "Usage: " + QUERY );
249                            
250                            samples.demoQuery();
251                    }
252                    else if ( cmd.equals( LIST_ATTRIBUTES ) )
253                    {
254                            if ( numArgs == 0 )
255                            {
256                                    samples.displayAllAttributes( getDomainRoot() );
257                            }
258                            else
259                            {
260                                    for( int i = 1; i < parts.length; ++i )
261                                    {
262                                            samples.displayAllAttributes( parts[ i ] );
263                                    }
264                            }
265                    }
266                    else if ( cmd.equals( LIST_DOTTED_NAMES ) )
267                    {
268                            require( numArgs == 0, "Usage: " + LIST_DOTTED_NAMES );
269                            samples.displayDottedNames( );
270                    }
271                    else if ( cmd.equals( LIST_PROPERTIES ) )
272                    {
273                            require( numArgs == 0, "Usage: " + LIST_PROPERTIES );
274                            samples.displayAllProperties( );
275                    }
276                    else if ( cmd.equals( RUN_ALL ) )
277                    {
278                            require( numArgs == 0, "Usage: " + RUN_ALL );
279                            
280                            for( int i = 0; i < RUN_ALL_CMDS.length; ++i )
281                            {
282                                    final String    choice  = RUN_ALL_CMDS[ i ];
283                                    
284                                    handleChoice( samples, choice );
285                            }
286                    }
287                    else if ( cmd.equals( DEMO_JMX_MONITOR ) )
288                    {
289                            samples.demoJMXMonitor();
290                    }
291                    else if ( cmd.equals( SET_MONITORING ) )
292                    {
293                            require( numArgs == 2, "Usage: " + SET_MONITORING + " <config-name> HIGH|LOW|OFF" );
294                            
295                            samples.setMonitoring( args[ 0 ], args[ 1 ]);
296                    }
297                    else if ( cmd.equals( DISPLAY_LOGGING ) )
298                    {
299                            samples.displayLogging( args );
300                    }
301                    else if ( cmd.equals( LISTEN_TO_LOGGING ) )
302                    {
303                            require( numArgs == 1, "Usage: " + LISTEN_TO_LOGGING + " true|false [server]*" );
304                            
305                            final String[] serverNames  = ArrayUtil.newArray( args, 1, args.length - 1 );
306                            samples.listenToLogging( new Boolean( args[ 0 ] ), serverNames );
307                    }
308                    else
309                    {
310                        handleUnknownCmd( parts );
311                    }
312            }

315                private void
316        handleUnknownCmd( final String[] parts )
317        {
318                    final String    cmd     = parts[ 0 ];
319                final Class[]    sig    = new Class[]
320                {
321                    DomainRoot.class,
322                    String[].class
323                };
324                
325                Class  cmdClass = null;
326                try
327                {
328                        try
329                        {
330                            cmdClass = Class.forName( cmd );
331                        }
332                        catch( Exception e )
333                        {
334                            cmdClass = Class.forName( this.getClass().getPackage().getName() + "." + cmd );
335                        }
336                }
337                catch( Throwable t )
338                {
339                        SampleUtil.println( "Unknown command or class not found: " + cmd );
340                        SampleUtil.println( COMMANDS );
341                }
342                
343                
344                if ( cmdClass != null )
345                {
346                try
347                {
348                    final Object cmdImpl      = cmdClass.newInstance();
349                    final Method runMethod  = cmdClass.getMethod( "runCmd", sig );
350                    
351                    final Object[] args = new Object[] { mDomainRoot, parts };
352                    final Object   result =runMethod.invoke( cmdImpl, args );
353                    SampleUtil.println( SmartStringifier.DEFAULT.stringify( result ) );
354                }
355                catch( Throwable t )
356                {
357                    final String msg   = ExceptionUtil.toString( t );
358                        SampleUtil.println( "ERROR:" );
359                        SampleUtil.println( msg );
360                }
361                }
362        }

364                    private void
365            demo()
366                    throws IOException
367            {
368                    final LineReaderImpl    in      = new LineReaderImpl( System.in );
369                    
370                    final Samples   samples = new Samples( getDomainRoot() );
371                    while ( true )
372                    {
373                            final String    line    = in.readLine( "\n" + PROMPT );
374                            try
375                            {
376                                    handleChoice( samples, line.trim() );
377                            }
378                            catch( IllegalArgumentException e )
379                            {
380                            }
381                            catch( Exception e )
382                            {
383                                    e.printStackTrace();
384                            }
385                    }
386            }

397                    private TLSParams
398            createTLSParams(
399                    final String    trustStore,
400                    final String    password )
401            {
402                    final File trustStoreFile       = new File( trustStore );
403                    final char[] trustStorePassword = password.toCharArray();
404                                            
405                    mHandshakeCompletedListener     = new HandshakeCompletedListenerImpl();
406                    final TrustStoreTrustManager trustMgr =
407                            new TrustStoreTrustManager( trustStoreFile, trustStorePassword);
408                    trustMgr.setPrompt( true );
409    
410                    final TLSParams tlsParams = new TLSParams( trustMgr, mHandshakeCompletedListener );
411    
412                    return( tlsParams );
413            }

415            /**
416                    Read connect properties from a file.
417             */
418                    private final Properties
419            getConnectProperties( final String file )
420                    throws IOException
421            {
422                    final Properties        props   = new Properties();
423                    
424                    if ( file != null )
425                    {
426                            SampleUtil.println( "Reading properties from: " + StringUtil.quote( file ) );
427                            final File      f       = new File( file );
428                            
429                            if ( f.exists() )
430                            {
431                                    final FileInputStream   is      = new FileInputStream( f );
432                                    try
433                                    {
434                                            props.load( is );
435                                    }
436                                    finally
437                                    {
438                                            is.close();
439                                    }
440                            }
441                            else
442                            {
443                                    SampleUtil.println("File \"" + file + " does not exist, using defaults." );
444                            }
445                    }
446                    
447                    return( props );
448            }

450            private final static String     DEFAULT_TRUST_STORE_FILE                = "~/.keystore";
451            private final static String     DEFAULT_TRUST_STORE_PASSWORD    = "changeme";
452            
453            /**
454                    @param host     hostname or IP address of Domain Admin Server
455                    @param port     RMI administrative port
456                    @param user     admin user
457                    @param password admin user password
458                    @param tlsParams TLS parameters, may be null
459                    @return AppserverConnectionSource
460             */
461                    public static AppserverConnectionSource
462            connect(
463                    final String    host,
464                    final int               port,
465                    final String    user,
466                    final String    password,
467                    final TLSParams tlsParams )
468                    throws IOException
469            {
470                    final String info = "host=" + host + ", port=" + port +
471                            ", user=" + user + ", password=" + password +
472                            ", tls=" + (tlsParams != null);
473                            
474                    SampleUtil.println( "Connecting...:" + info );
475                    
476                    final AppserverConnectionSource conn    =
477                            new AppserverConnectionSource( AppserverConnectionSource.PROTOCOL_RMI,
478                                    host, port, user, password, tlsParams, null);
479                    
480                    // force the connection now
481                    conn.getJMXConnector( false );
482    
483                    SampleUtil.println( "Connected: " + info );
484                    
485                    return( conn );
486            }

489                private AppserverConnectionSource
490             attemptConnect(
491                    final String    host,
492                    final int               port,
493                    final String    user,
494                    final String    password,
495                final boolean  useTLS,
496                final String   trustStore,
497                final String   trustStorePassword )
498                    throws IOException
499             {
500                final TLSParams     tlsParams       = useTLS ?
501                            createTLSParams( trustStore, trustStorePassword) : null;
502                    
503                    final AppserverConnectionSource conn    = connect( host, port, user, password, tlsParams );
504                    
505                    if ( useTLS && mHandshakeCompletedListener != null )
506                    {
507                            SampleUtil.println( "HandshakeCompletedEvent: " +
508                                    SmartStringifier.toString( mHandshakeCompletedListener.getLastEvent() ) );
509                    }
510                    
511                    return conn;
512             }

514            /**
515             */
516                    public
517            SampleMain( final String optionalPropertiesFile )
518                    throws IOException
519            {
520                    final Properties        props   = getConnectProperties( optionalPropertiesFile );
521                    
522                    final String    host    = props.getProperty( "connect.host", "localhost" );
523                    final int               port    = Integer.parseInt( props.getProperty( "connect.port", "8686" ) );
524                    final String    user    = props.getProperty( "connect.user", "admin" );
525                    final String    password        = props.getProperty( "connect.password", "admin123" );
526                    final String    trustStore      = props.getProperty( "connect.truststore", DEFAULT_TRUST_STORE_FILE);
527                    final String    trustStorePassword      =
528                                    props.getProperty( "connect.truststorePassword", DEFAULT_TRUST_STORE_PASSWORD);
529                    final boolean   useTLS  = 
530                            Boolean.valueOf( props.getProperty( "connect.useTLS", "false" ) ).booleanValue();
531                    
532                    AppserverConnectionSource       conn    = null;
533                    try
534                    {
535                        conn    = attemptConnect( host, port, user, password,
536                                    useTLS, trustStore, trustStorePassword );
537                    }
538                    catch( Throwable t )
539                    {
540                        final Throwable rootCause   = ExceptionUtil.getRootCause( t );
541                        SampleUtil.println( "Caught " + rootCause.getClass().getName() );
542                        
543                        if ( rootCause instanceof CertificateException  )
544                        {
545                            SampleUtil.println( "Attempting connection with connect.useTLS = " + ! useTLS );
546                        conn    = attemptConnect( host, port, user, password,
547                                    ! useTLS, trustStore, trustStorePassword );
548                        }
549                    }
550                    
551                    assert conn != null : "Can't connect to server";
552                    mDomainRoot     = conn.getDomainRoot();
553                    
554                    try
555                    {
556                            demo( );
557                    }
558                    finally
559                    {
560                            // close the connection (not necessary, but here for as an example)
561                            conn.getJMXConnector( false ).close();
562                    }
563            }
