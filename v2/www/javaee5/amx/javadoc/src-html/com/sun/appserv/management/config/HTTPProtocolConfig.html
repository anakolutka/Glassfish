031    package com.sun.appserv.management.config;
032    
033    import com.sun.appserv.management.base.XTypes;
034    
035    /**
036    Configuration for the &lt;http-protocol&gt; element.
037    */
038    
039    public interface HTTPProtocolConfig extends ConfigElement 
040    {
041    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
042            public static final String      J2EE_TYPE       = XTypes.HTTP_PROTOCOL_CONFIG;
043    
044            public String   getDefaultResponseType();
045            /**
046                    Legal values include:
047                    <ul>
048                    <li>"text/html</li>
049                    <li>ISO-8859-1</li>
050                    <li>en</li>
051                    </ul>
052            */
053            public void     setDefaultResponseType( final String value );
054    
055            public boolean  getDNSLookupEnabled();
056            public void     setDNSLookupEnabled( final boolean value );
057    
058            /**
059                    @see HTTPProtocolConfigKeys#FORCED_RESPONSE_TYPE_KEY
060            */
061            public String   getForcedResponseType();
062            /**
063                    @see HTTPProtocolConfigKeys#FORCED_RESPONSE_TYPE_KEY
064            */
065            public void     setForcedResponseType( final String value );
066    
067            public boolean  getSSLEnabled();
068            public void     setSSLEnabled( final boolean value );
069    
070            public String   getVersion();
071            public void     setVersion( final String value );
072    
073    }
