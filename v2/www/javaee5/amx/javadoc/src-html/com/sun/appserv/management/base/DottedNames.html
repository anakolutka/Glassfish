023    package com.sun.appserv.management.base;
024    
025    import com.sun.appserv.management.base.XTypes;
026    
027    /**
028            Programmatic access to dotted-names corresponding to <i>asadmin</i> CLI
029            functionality.
030            <p>
031            Note that the set of dotted names is periodically refreshed; to force
032            an update use {@link #refresh}.
033            
034            @see DottedNames
035            @see AllDottedNames
036            @see com.sun.appserv.management.config.ConfigDottedNames
037            @see com.sun.appserv.management.monitor.MonitoringDottedNames
038     */
039    public interface DottedNames extends AMX
040    {
041            /**
042                    Return an array of values corresponding to each dotted-name.
043                    Each slot in the array will contain either an Attribute or an Exception.
044             */
045            public Object[] dottedNameGet( String[] names );
046            
047            /**
048                    Return a value for a dotted-name.  If a name does not exist,
049                    then null is returned.
050                    @param name
051             */
052            public Object   dottedNameGet( String name );
053            
054            /**
055                    List all valid prefixes for dotted names
056                    @param names
057             */
058            public Object[] dottedNameList( String[] names );
059            
060            
061            /**
062                    Set values for dotted names; each entry must be of the form:
063                    <pre>
064                <i>dotted-name</i>=<i>value</i>
065            </pre> 
066                    @param nameValuePairs
067             */
068            public Object[] dottedNameSet( String[] nameValuePairs );
069            
070            /**
071                    Force a refresh of the list of dotted names.
072             */
073            public void     refresh();
074    
075    
076    
077    }
