030    package com.sun.appserv.management.config;
031    
032    import com.sun.appserv.management.base.XTypes;
033    
034    
035    
036    
037    /**
038             Configuration for the &lt;jdbc-connection-pool&gt; element.
039    */
040    
041    public interface JDBCConnectionPoolConfig
042        extends NamedConfigElement, Description, PropertiesAccess, ResourceRefConfigReferent
043    {
044    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
045            public static final String      J2EE_TYPE                       = XTypes.JDBC_CONNECTION_POOL_CONFIG;
046            
047            public String   getConnectionValidationMethod();
048            public void     setConnectionValidationMethod( String value );
049    
050            public String   getDatasourceClassname();
051            public void     setDatasourceClassname( String value );
052    
053            public boolean  getFailAllConnections();
054            public void     setFailAllConnections( boolean value );
055    
056            public String   getIdleTimeoutInSeconds();
057            public void     setIdleTimeoutInSeconds( String value );
058    
059            public boolean  getIsConnectionValidationRequired();
060            public void     setIsConnectionValidationRequired( boolean value );
061    
062            public boolean  getIsIsolationLevelGuaranteed();
063            public void     setIsIsolationLevelGuaranteed( boolean value );
064    
065            public String   getMaxPoolSize();
066            public void     setMaxPoolSize( String value );
067    
068            public String   getMaxWaitTimeInMillis();
069            public void     setMaxWaitTimeInMillis( String value );
070    
071            public String   getPoolResizeQuantity();
072            public void     setPoolResizeQuantity( String value );
073    
074            public String   getResType();
075            public void     setResType( String value );
076    
077            public String   getSteadyPoolSize();
078            public void     setSteadyPoolSize( String value );
079    
080            public String   getTransactionIsolationLevel();
081            /**
082                    See {@link IsolationValues}.
083             */
084            public void     setTransactionIsolationLevel( String value );
085    
086            public String   getValidationTableName();
087            public void     setValidationTableName( String value );
088    
089        /**                            
090            A pool with this property set to true returns                 
091            non-transactional connections. This connection does not get   
092            automatically enlisted with the transaction manager.
093            
094            @since AppServer 9.0
095         */
096        public boolean getNonTransactionalConnections();
097        
098        /**
099            @see #getNonTransactionalConnections          
100            @since AppServer 9.0
101         */
102        public void setNonTransactionalConnections( boolean enabled );
103        
104        /**                                
105            A pool with this property set to true, can be used by         
106            non-J2EE components (i.e components other than EJBs or        
107            Servlets). The returned connection is enlisted automatically  
108            with the transaction context obtained from the transaction    
109            manager. This property is to enable the pool to be used by    
110            non-component callers such as ServletFilters, Lifecycle       
111            modules, and 3rd party persistence managers. Standard J2EE    
112            components can continue to use such pools. Connections        
113            obtained by non-component callers are not automatically       
114            cleaned at the end of a transaction by the container. They    
115            need to be explicitly closed by the the caller.               
116            
117            @since AppServer 9.0
118         */
119        public boolean getAllowNonComponentCallers();
120        
121        /**
122            @see #getAllowNonComponentCallers          
123            @since AppServer 9.0
124         */
125        public void setAllowNonComponentCallers( boolean enabled );
126    
127    
128    
129    }
