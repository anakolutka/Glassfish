029    package com.sun.appserv.management.base;
030    
031    import javax.management.Attribute;
032    
033    import com.sun.appserv.management.base.AMX;
034    import com.sun.appserv.management.base.XTypes;
035    
036    /**
037            Interface for a sample MBean , used as target for sample and test code.
038            Various Attributes of varying types are made available for testing.
039     */
040    public interface Sample extends Utility, AMX, Singleton
041    {
042    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
043            public static final String      J2EE_TYPE                       = XTypes.SAMPLE;
044            
045            /**
046                    The type of Notification emitted by emitNotification().
047             */
048            public static final String      SAMPLE_NOTIFICATION_TYPE        = "Sample";
049            
050            /**
051                    The key to access user data within the Map obtained from Notification.getUserData().
052             */
053            public static final String      USER_DATA_KEY                           = "UserData";
054            
055            /**
056                    Emit 'numNotifs' notifications of type
057                    SAMPLE_NOTIFICATION_TYPE at the specified interval.
058                    
059                    @param data arbitrary data which will be placed into the Notification's UserData field.
060                    @param numNotifs number of Notifications to issue >= 1
061                    @param intervalMillis interval at which Notifications should be issued >= 0
062             */
063            public void     emitNotifications( final Object data, final int numNotifs, final long intervalMillis );
064            
065            /** 
066                    Add a new Attribute. After this, the MBeanInfo will contain an MBeanAttributeInfo
067                    for this Attribute.
068                    
069                    @param name
070                    @param value
071             */
072            public void     addAttribute( final String name, final Object value );
073            
074            /** 
075                    Remove an Attribute. After this, the MBeanInfo will no longer
076                    contain an MBeanAttributeInfo for this Attribute.
077             */
078            public void     removeAttribute( final String name );
079            
080            /**
081                    For testing bandwidth...
082             */
083            public void             uploadBytes( final byte[] bytes );
084            public byte[]   downloadBytes( final int numBytes );
085    }
