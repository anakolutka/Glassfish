023    package com.sun.appserv.management.monitor;
024    
025    import javax.management.openmbean.CompositeDataSupport;
026    import javax.management.j2ee.statistics.Stats;
027    import javax.management.j2ee.statistics.Statistic;
028    
029    /**
030            Monitoring MBeans which expose one or more
031            {@link Statistic} implement this interface.
032            <p>
033            Each MonitoringStats MBean exposes Statistics as Attributes formed
034            using the following pattern:
035    <pre>    <i>statistic-name</i>_<i>field-name</i> </pre>
036            For example, a CountStatistic names "Hosts" would generate the following
037            Attributes:
038            <ul>
039            <li>Hosts_Name</li>
040            <li>Hosts_Description</li>
041            <li>Hosts_Unit</li>
042            <li>Hosts_StartTime</li>
043            <li>Hosts_LastSampleTime</li>
044            <li>Hosts_Count</li>
045            </ul>
046            For most purposes, it is the "<i>name</i>_Count" value which should be of
047            primary interest.
048            <p>
049            The type of the Statistic will govern how many Attributes are generated, 
050            based on its actual fields, but every Statistic interface will have
051            at least the Name, Description, Unit, StartTime and LastSampleTime fields present.
052        <p>
053        <b>Note:</b> These derived Attributes are not made available directly
054        in the interface for the MBean</b>; please refer to the documentation
055        for each MonitoringStats MBean to determine which Statistics are available,
056        and thus which derived Attributes available.
057        @see com.sun.appserv.management.base.AMX
058        @see com.sun.appserv.management.base.StdAttributesAccess
059        @see com.sun.appserv.management.base.Util#getExtra
060        @see com.sun.appserv.management.base.Extra
061        @see javax.management.j2ee.statistics.Statistic
062     */
063    public interface MonitoringStats extends Monitoring
064    {
065            /**
066                    Get a JSR 77 Stats object for all available statistics. Semantically,
067                    the returned Stats object MUST be a "snapshot" to the current state
068                    of the world.
069             */
070            public Stats            getStats();
071            public String           getStatsInterfaceName();
072            
073            /**
074                    Get a specific JSR 77 Statistic.
075                    
076                    @param name     the Statistic name
077             */
078            public Statistic        getStatistic( String name );
079            
080            /**
081                    Get specific JSR 77 Statistics. If a Statistic is not found, then null
082                    is returned in its array slot.
083                    
084                    @param names    the Statistic names
085                    @return corresponding values for the names
086             */
087            public Statistic[]      getStatistics( String[] names );
088            
089            /**
090                    Get the names of all available Statistics.
091             */
092            public String[] getStatisticNames( );
093            
094            
095            /**
096                    Get a JSR 77 Stats object encoded as a standard serializable JMX OpenType.
097             */
098            public CompositeDataSupport             getOpenStats();
099            
100            
101            /**
102                    Get a JSR 77 Statistic encoded as a standard serializable JMX OpenType.
103                    
104                    @param name     the Statistic name
105             */
106            public CompositeDataSupport             getOpenStatistic( String name );
107            
108            
109            /**
110                    Get JSR 77 Statistics encoded as a standard serializable JMX OpenTypes.
111                    If a Statistic is not found, then null is returned in its array slot.
112                    
113                    @param names    the Statistic names
114                    @return corresponding values for the names
115             */
116            public CompositeDataSupport[]           getOpenStatistics( String[] names );
117            
118            
119            /**
120                    Refresh any stale data.  This may or may not be applicable, depending
121                    on how the implementation performs its data collection.  If the data
122                    is always "live", then the implementation of this routine does nothing.
123                    
124                    @return true if there may be new data present, false otherwise
125             */
126            public boolean          refresh();
127    }
