040    /**
041             Configuration for the &lt;session-manager&gt; element.
042     */
043    public interface SessionManagerConfig
044            extends ConfigElement, Container

049            /**
050                    Creates new manager-properties element.
051             
052                    @param params Map of optional attributes whose keys are defined in
053                    ManagerPropertiesParams interface.
054                    @return A proxy to the ManagerPropertiesConfig MBean.
055                    @see ManagerPropertiesConfigKeys
056             */
057            public ManagerPropertiesConfig createManagerPropertiesConfig( Map<String,String> params );
058    
059            /**
060                    Removes manager-properties element.
061             */
062            public void removeManagerPropertiesConfig();
063    
064            /**
065                    Get the ManagerPropertiesConfig MBean.
066             */
067            public ManagerPropertiesConfig getManagerPropertiesConfig();
068    
069            /**
070                    Creates new manager-properties element.
071             
072                    @param params Map of optional attributes whose keys are defined in
073                    ManagerPropertiesParams interface.
074                    @return A proxy to the StorePropertiesConfig MBean.
075                    @see StorePropertiesConfigKeys
076             */
077            public StorePropertiesConfig createStorePropertiesConfig( Map<String,String> params );
078    
079            /**
080                    Removes store-properties element.
081             */
082            public void removeStorePropertiesConfig();
083    
084            /**
085                    Get the StorePropertiesConfig MBean.
086             */
087            public StorePropertiesConfig getStorePropertiesConfig();
088    }
