023    package com.sun.appserv.management.client.prefs;
024    import java.util.Collection;
025    
026    /**
027     * An interface that represents the database of LoginInfo objects. Provides methods
028     * to access and mutate the store. The general contract of store update comprises of the following:
029     * <ul>
030     *   <li> The store can contain at the most one LoginInfo for a host and port combination </li>
031     *   <li> There is <i> no guarantee </i> that concurrent modification of this store by two
032     *        different programs will always be consistent. </li>
033     * </ul>
034     * @since Appserver 9.0
035     */
036    public interface LoginInfoStore {
037        
038        /**
039         * Returns a {@link LoginInfo} corresponding to the given host and port, from this store.
040         * The host may not be null. For a given host and port, there can be at most
041         * one LoginInfo in this store.
042         * @param host a non null String representing host name
043         * @param port an integer specifying the port number
044         * @return the corresponding LoginInfo, null if there is none
045         * @throws StoreException if there is something wrong with reading the store
046         * @throws IllegalArgumentException if the parameter host is null
047         */
048        public LoginInfo read(final String host, final int port) throws StoreException;
049        
050        /**
051         * Removes the {@link LoginInfo} corresponding to the given host and port, from this store.
052         * The host may not be null. If no such LoginInfo exists, StoreException results.
053         * The caller thus must ensure if such a LoginInfo exists before calling this method. 
054         * Upon successful return, size of this store decreases by one.
055         * @param host a non null String representing host name
056         * @param port an integer specifying the port number
057         * @throws StoreException if there is something wrong with reading the store or if there is
058         *         no such LoginInfo
059         * @throws IllegalArgumentException if the parameter host is null
060         */
061        public void remove(final String host, final int port) throws StoreException;
062        
063        /**
064         * Stores the given LoginInfo in this store. Given LoginInfo may not be null.
065         * Upon successful return, the size of this store increases by one. An exception is thrown
066         * if there is already a LoginInfo with given host and port.
067         * @param login a LoginInfo that needs to be stored
068         * @throws StoreException if there's any problem or if there is already a LoginInfo
069         * with given host and port
070         * @throws IllegalArgumentException if the given LoginInfo is null
071         */
072        public void store(final LoginInfo login) throws StoreException;
073        
074        /**
075         * Stores the given LoginInfo in this store. Given LoginInfo may not be null.
076         * Upon successful return, the size of this store increases by one. An exception is thrown
077         * if there is already a LoginInfo with given host and port and overwrite is false.
078         * If overwrite is true, the given LoginInfo is stored regardless of whether it already
079         * exists in this store. Depending upon the value of overwrite, the store is either unchanged
080         * or not.
081         * @param login a LoginInfo that needs to be stored
082         * @throws StoreException if there's any problem in storing or if overwrite is false and
083         * the LoginInfo with given host and port already exists
084         * @throws IllegalArgumentException if the given LoginInfo is null
085         */
086        public void store(final LoginInfo login, final boolean overwrite) throws StoreException;
087        
088       /**
089         * Checks whether a LoginInfo for given host and port exists in this store.
090         * @param host a non null String representing host name
091         * @param port an integer specifying the port number
092         * @throws StoreException if there's any problem reading the store
093         */
094        public boolean exists(final String host, final int port) throws StoreException;
095        
096        /**
097         * A convenience method that returns the Collection of LoginInfo instances stored in this store.
098         * An empty Collection is returned when there are no LoginInfo items stored.
099         * @return the Collection of LoginInfo instances
100         * @throws StoreException if there's any problem reading the store
101         */
102        public Collection<LoginInfo> list() throws StoreException;
103        
104        /**
105         * A convenience method that returns the number of LoginInfo instances stored in this store.
106         * Zero is returned when no login information is stored.
107         * @return an integer representing number of stored login information elements, 0 if none
108         * @throws StoreException if there's any problem reading the store
109         */
110        public int size() throws StoreException;
111        
112        /** Returns the name of the store. 
113         * This is any name that the store implementation wants to use for identification, for instance.
114         */
115        public String getName();
116    }
