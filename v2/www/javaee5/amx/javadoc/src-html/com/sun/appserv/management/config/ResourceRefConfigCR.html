030    package com.sun.appserv.management.config;
031    
032    import java.util.Map;
033    
034    /**
035            Allows creation and removal of ResourceRefConfig.
036    */
037    public interface ResourceRefConfigCR extends ConfigRemover
038    {
039            /**
040                    Creates a new resource-ref element for the given target
041                    @param referencedResourceName resource name
042                    @param optional Map of optional attributes.
043                    @return A proxy to the ResourceRefConfig MBean.
044             */
045            public ResourceRefConfig createResourceRefConfig(String referencedResourceName,
046                        Map<String,String> optional);
047                
048            /**
049                    Creates a new resource-ref element for the given target
050                    @param referencedResourceName resource name
051                    @return A proxy to the ResourceRefConfig MBean.
052             */
053            public ResourceRefConfig createResourceRefConfig(String referencedResourceName);
054    
055    
056            /**
057                    Creates a new resource-ref element.
058                    @param referencedResourceName resource name
059                    @param enabled
060                    @return A proxy to the ResourceRefConfig MBean.
061             */
062            public ResourceRefConfig createResourceRefConfig(String referencedResourceName, 
063                    boolean enabled);
064    
065            /**
066             */
067            public void     removeResourceRefConfig( String name );
068    }
