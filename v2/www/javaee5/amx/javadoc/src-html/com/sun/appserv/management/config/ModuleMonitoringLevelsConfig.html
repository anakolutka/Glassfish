031    package com.sun.appserv.management.config;
032    
033    import java.util.Map;
034    
035    import com.sun.appserv.management.base.XTypes;
036    
037    
038    
039    /**
040            Configuration for the &lt;module-monitoring-levels&gt; element.
041            
042            Each monitoring level can be one of the values defined in
043            {@link ModuleMonitoringLevelValues}.
044     */
045    public interface ModuleMonitoringLevelsConfig extends ConfigElement, PropertiesAccess
046    {
047    /** The j2eeType as returned by {@link com.sun.appserv.management.base.AMX#getJ2EEType}. */
048            public static final String      J2EE_TYPE       = XTypes.MODULE_MONITORING_LEVELS_CONFIG;
049            
050            /**
051                    Get a Map keyed by Attribute name of all monitoring levels.
052                    Attribute names correspond to the various get methods eg
053                    "JVM" for {@link #getJVM},
054                    "ConnectorService" for {@link #getConnectorService}, etc.
055                    <p>
056                    The possible levels are as defined in {@link ModuleMonitoringLevelValues}.
057             */
058            public Map<String,String>         getAllLevels();
059            
060            /**
061                    Change all monitoring levels to the specified value.
062                    
063                    @param value one of the values in {@link ModuleMonitoringLevelValues}
064             */
065            public void             changeAll( final String value );
066            
067            public String   getJVM();
068            
069            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
070            public void             setJVM( final String value );
071            
072            public String   getConnectorService();
073            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
074            public void             setConnectorService( final String value );
075            
076            public String   getJMSService();
077            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
078            public void             setJMSService( final String value );
079            
080            public String   getConnectorConnectionPool();
081            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
082            public void     setConnectorConnectionPool( final String value );
083    
084            public String   getEJBContainer();
085            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
086            public void     setEJBContainer( final String value );
087    
088            public String   getHTTPService();
089            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
090            public void     setHTTPService( final String value );
091    
092            public String   getJDBCConnectionPool();
093            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
094            public void     setJDBCConnectionPool( final String value );
095    
096            public String   getORB();
097            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
098            public void     setORB( final String value );
099    
100            public String   getThreadPool();
101            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
102            public void     setThreadPool( final String value );
103    
104            public String   getTransactionService();
105            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
106            public void     setTransactionService( final String value );
107    
108            public String   getWebContainer();
109            /** @param value one of the values in {@link ModuleMonitoringLevelValues} */
110            public void     setWebContainer( final String value );
111    
112    
113    
114    
115    }
