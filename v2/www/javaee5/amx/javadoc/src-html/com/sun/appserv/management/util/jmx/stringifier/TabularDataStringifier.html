023    package com.sun.appserv.management.util.jmx.stringifier;
024    
025    import java.util.Set;
026    import java.util.List;
027    import java.util.Iterator;
028    
029    import javax.management.openmbean.TabularData;
030    import javax.management.openmbean.CompositeData;
031    import javax.management.openmbean.TabularType;
032    
033    import com.sun.appserv.management.util.misc.StringUtil;
034    import com.sun.appserv.management.util.stringifier.Stringifier;
035    import com.sun.appserv.management.util.stringifier.SmartStringifier;
036    
037    
038    public class TabularDataStringifier implements Stringifier
039    {
040            public static final TabularDataStringifier      DEFAULT = new TabularDataStringifier();
041            
042                    public
043            TabularDataStringifier( )
044            {
045            }
046            
047                    public String
048            stringify( Object o )
049            {
050                    final StringBuffer      buf     = new StringBuffer();
051                    buf.append( "Tabular data:\n" );
052                    
053                    final TabularData       data    = (TabularData)o;
054                    final TabularType       type    = data.getTabularType();
055                    
056                    final List              indexNames      = type.getIndexNames();
057                    final Set               rowKeys         = data.keySet();
058                    final Iterator  rowIter = rowKeys.iterator();
059                    int                             rowIndex        = 0;
060                    while ( rowIter.hasNext() )
061                    {
062                            final Object[]          key             = (Object[])rowIter.next();
063                            final CompositeData     item    = data.get( key );
064                            
065                            final String    s       = SmartStringifier.toString( item );
066                            
067                            // emit the row index followed by the row
068                            buf.append( "[" + rowIndex + "] " );
069                            buf.append( s + "\n" );
070                            
071                            ++rowIndex;
072                    }
073                    
074                    return( buf.toString() );
075            }
076    }
