030    package com.sun.appserv.management.monitor.statistics;
031    
032    import javax.management.j2ee.statistics.Stats;
033    
034    import com.sun.appserv.management.j2ee.statistics.StringStatistic;
035    
036    /**
037            @see com.sun.appserv.management.monitor.HTTPServiceVirtualServerMonitor
038     */
039    public interface HTTPServiceVirtualServerStats extends Stats
040    {
041            /**
042                    Current => Space-separated list of IP addresses
043             */
044            public StringStatistic getInterfaces();
045            
046            /**
047                    Current => Space-separated list of hosts
048             */
049            public StringStatistic getHosts();
050            
051            /**
052                    Current one of:
053                    <ul>
054                    <li>"active"</li>
055                    <li>"off"</li>
056                    <li>"disabled"</li>
057                    </ul>
058             */
059            public StringStatistic getMode();
060            
061            /**
062                    Current => server name.
063             */
064            public StringStatistic getID();
065    }
