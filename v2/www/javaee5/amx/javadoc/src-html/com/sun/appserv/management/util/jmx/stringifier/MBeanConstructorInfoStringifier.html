023    package com.sun.appserv.management.util.jmx.stringifier;
024    
025    import java.lang.reflect.Array;
026    import javax.management.MBeanConstructorInfo;
027    
028    import com.sun.appserv.management.util.stringifier.Stringifier;
029    
030    
031    public class MBeanConstructorInfoStringifier extends MBeanFeatureInfoStringifier implements Stringifier
032    {
033            public static final MBeanConstructorInfoStringifier     DEFAULT =
034                new MBeanConstructorInfoStringifier();
035    
036    
037                    public
038            MBeanConstructorInfoStringifier()
039            {
040                    super( );
041            }
042            
043                    public
044            MBeanConstructorInfoStringifier( MBeanFeatureInfoStringifierOptions options )
045            {
046                    super( options );
047            }
048            
049                    public String
050            stringify( Object o )
051            {
052                    final MBeanConstructorInfo      constructor     = (MBeanConstructorInfo)o;
053                    
054                    final String    name    = constructor.getName();
055                    final int               lastDot =       name.lastIndexOf( "." );
056                    final String    abbreviatedName = name.substring( lastDot + 1, name.length() );
057                    
058                    final String params     = "(" +
059                            ParamsToString( constructor.getSignature(), mOptions ) + ")";
060                    
061                    return( abbreviatedName + params );
062            }
063    }
